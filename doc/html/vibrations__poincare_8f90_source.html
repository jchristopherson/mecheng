5 ! ------------------------------------------------------------------------------
6  module function compute_poincare_section(t, x, v, period, err) result(rst)
7  ! Arguments
8  real(real64), intent(in), dimension(:) :: t, x, v
9  real(real64), intent(in) :: period
10  class(errors), intent(inout), optional, target :: err
11  real(real64), allocatable, dimension(:,:) :: rst
12 
13  ! Local Variables
14  class(errors), pointer :: errmgr
15  type(errors), target :: deferr
16  character(len = 256) :: errmsg
17  integer(int32) :: i, npts, nsample, flag
18  real(real64) :: maxt, mint
19  real(real64), allocatable, dimension(:) :: samples
20  type(linear_interp) :: interp
21 
22  ! Initialization
23  if (present(err)) then
24  errmgr => err
25  else
26  errmgr => deferr
27  end if
28  npts = size(t)
29 
30  ! Ensure the input arrays are properly sized
31  if (size(x) /= npts) then
32  write(errmsg, '(AI0AI0A)') &
33  "The input array was expected to be of size ", &
34  npts, ", but was found to be of size ", size(x), "."
35  call errmgr%report_error("compute_poincare_section", &
36  trim(errmsg), mech_array_size_error)
37  return
38  end if
39  if (size(v) /= npts) then
40  write(errmsg, '(AI0AI0A)') &
41  "The derivative array was expected to be of size ", &
42  npts, ", but was found to be of size ", size(v), "."
43  call errmgr%report_error("compute_poincare_section", &
44  trim(errmsg), mech_array_size_error)
45  return
46  end if
47 
48  ! Ensure period is a positive, non-zero value
49  if (period < epsilon(period)) then
50  call errmgr%report_error("compute_poincare_section", &
51  "The sampling period must be a positive value larger " // &
52  "than machine precision.", mech_invalid_input_error)
53  return
54  end if
55 
56  ! Build the time array at which values are to be sampled
57  maxt = maxval(t)
58  mint = minval(t)
59  nsample = int((maxt - mint) / period, int32) + 1
60  allocate(samples(nsample), stat = flag)
61  if (flag == 0) allocate(rst(nsample, 2), stat = flag)
62  if (flag /= 0) then
63  call errmgr%report_error("compute_poincare_section", &
64  "Insufficient memory available.", &
65  mech_out_of_memory_error)
66  return
67  end if
68 
69  samples(1) = mint
70  do i = 2, nsample
71  samples(i) = samples(i-1) + period
72  end do
73 
74  ! Interpolate to obtain the values - use linear interpolation
75  call interp%initialize(t, x, err = errmgr)
76  if (errmgr%has_error_occurred()) return
77  rst(:,1) = interp%interpolate(samples)
78 
79  call interp%initialize(t, v, err = errmgr)
80  if (errmgr%has_error_occurred()) return
81  rst(:,2) = interp%interpolate(samples)
82  end function
