150  ! Takes a single step of the fitting algorithm.
151  !

153  ! 1. B. Gustavsen and A. Semlyen, "Rational approximation of frequency
154  ! domain responses by Vector Fitting", IEEE Trans. Power Delivery,
155  ! vol. 14, no. 3, pp. 1052-1061, July 1999. Link
156  ! 2. B. Gustavsen, "Improving the pole relocating properties of vector
157  ! fitting", IEEE Trans. Power Delivery, vol. 21, no. 3, pp. 1587-1592,

159  ! 3. D. Deschrijver, M. Mrozowski, T. Dhaene, and D. De Zutter,

161  ! the Vector Fitting Method”, IEEE Microwave and Wireless Components
162  ! Letters, vol. 18, no. 6, pp. 383-385, June 2008.
163  !
164  ! Websites:
165  ! - https://www.sintef.no/vectfit

167  ! Arguments:
168  ! - f: An M-element array containing the complex-valued frequency response

170  ! - s: An M-element array containing the complex-valued frequency points at

172  ! - poles: An N-element array containing an estimate of the system poles.
173  ! On output, the new estimate of pole locations.
174  ! - weights: An M-element array containing weighting factors for each
175  ! frequency value.
176  ! - a [out]: An N -by- N matrix.
177  ! - b [out]: An N -by- 1 matrix.
178  ! - c [out]: A 1 -by- N matrix.
179  ! - d [out]: A 1 -by- 1 matrix.
180  ! - fit [out]: An M-element array containing the fitted data.
181  ! - delta [out]: An M-element array containing the difference between the
182  ! fitted and raw data.
183  ! - iwork: An X-element workspace array.
184  ! X = N
185  ! Breakdown:
186  ! - N - cindex
187  ! - cwork: An X-element workspace array.
188  ! X = N**2 + (3*M + 2)*N + 4*M
189  ! Breakdown:
190  ! - M-by-(N+1) - Dk
191  ! - M-by-2*(N+1) - Ac
192  ! - N - Cc
193  ! - N - Bc
194  ! - N-by-N - LAMBDA
195  ! - M - Bb
196  ! - dwork: An X-element workspace array.
197  ! X = (2*M + 1)*S + N**2 + (4*M + 5)*N + 6*M + 5

204  ! - N+1 - x
205  ! - 2*M - Br
206  ! - N+1-by-N+1 - At

327 ! ------------------------------------------------------------------------------
328  ! Identifies the real-valued and complex-conjugate pair values.
329  !
330  ! - vals: An N-element array of complex values to search.
331  ! - tol: A tolerance value to determine closeness to zero.
332  ! - labels [out]: An N-element array containing labels identifying each value
333  ! type. The labels are as follows:
334  ! * 0: Real
335  ! * 1: Complex
336  ! * 2: Complex Conjugate
337  subroutine label_complex_values(vals, tol, labels)
338  ! Arguments
339  complex(real64), intent(in), dimension(:) :: vals
340  real(real64), intent(in) :: tol
341  integer(int32), intent(out), dimension(:) :: labels
342 
343  ! Local Variables
344  integer(int32) :: i, n
345  real(real64) :: x, xc
346 
347  ! Process
348  n = size(vals)
349  i = 1
350  do while (i <= n)
351  x = aimag(vals(i))
352  if (abs(x) < tol) then
353  ! Real-Values
354  labels(i) = 0
355  i = i + 1
356  else
357  ! Complex-Valued
358  labels(i) = 1
359 
360  ! Check for the conjugate in the next index
361  if (i < n) then
362  xc = aimag(vals(i+1))
363  if (abs(x + xc) < tol) then
364  ! Complex-Conjugate
365  labels(i+1) = 2
366  i = i + 2
367  else
368  ! Not a complex-conjugate
369  i = i + 1
370  end if
371  else
372  ! Can't be a conjugate value as its the last value in the array
373  i = i + 1
374  end if
375  end if
376  end do
377  end subroutine

379 ! ------------------------------------------------------------------------------
380  ! Builds the denominator based upon the current pole location estimate.
381  !
382  ! - s: An M-element array containing the complex-valued frequency points at
383  ! which f is defined.
384  ! - poles: An N-element array containing an estimate of the system poles.
385  ! On output, the new estimate of pole locations.
386  ! - cindex: An N-element array determining what pole values are real,
387  ! complex, and/or complex-conjugates. See label_complex_values for
388  ! more info.
389  ! - dk [out]: An M-by-N+1 matrix.
390  subroutine build_denominator(s, poles, cindex, dk)
391  ! Arguments
392  complex(real64), intent(in), dimension(:) :: s, poles
393  integer(int32), intent(in), dimension(:) :: cindex
394  complex(real64), intent(out), dimension(:,:) :: dk
395 
396  ! Parameters
397  complex(real64), parameter :: one = (1.0d0, 0.0d0)
398  complex(real64), parameter :: j = (0.0d0, 1.0d0)
399 
400  ! Local Variables
401  integer(int32) :: i, ns, n
402 
403  ! Initialization
404  ns = size(s)
405  n = size(poles)
406  dk(:,n+1) = one
407 
408  ! Process
409  i = 1
410  do while (i <= n)
411  if (cindex(i) == 0) then
412  dk(:,i) = one / (s - poles(i))
413  i = i + 1
414  else if (cindex(i) == 1) then
415  dk(:,i) = one / (s - poles(i)) + one / (s - conjg(poles(i)))
416  dk(:,i+1) = j / (s - poles(i)) - j / (s - conjg(poles(i)))
417  i = i + 2
418  end if
419  end do
420  end subroutine

442  subroutine denom_to_state_space(scale, weights, f, dk, ac, ar, bc, br, &
443  tau, q, escale, x, a, b, c, d)
444  use linalg_core
445 
446  ! Arguments
447  real(real64), intent(in) :: scale
448  real(real64), intent(in), dimension(:) :: weights
449  complex(real64), intent(in), dimension(:) :: f
450  complex(real64), intent(in), dimension(:,:) :: dk
451  complex(real64), intent(out), dimension(:,:) :: ac
452  complex(real64), intent(out), dimension(:) :: bc
453  real(real64), intent(out), dimension(:,:) :: ar
454  real(real64), intent(out), dimension(:) :: tau, escale, x, br
455  real(real64), intent(out), dimension(:,:) :: q
456 
457  ! State Space Matrix Outputs
458  real(real64), intent(out), dimension(:,:) :: a, c, d
459  real(real64), intent(out), dimension(:) :: b
460 
461  ! Parameters
462  real(real64), parameter :: tol_low = 1.0d-18
463  real(real64), parameter :: tol_high = 1.0d18
464 
465  ! Local Variables
466  integer(int32) :: i, ns, n, offset, ind, ind1, ind2
467  real(real64) :: dnew
468 
469  ! Initialization
470  ns = size(dk, 1)
471  n = size(dk, 2) - 1 ! DK is Ns -by- (N + 1)
472  offset = n + 1;
473 
474  ! Process
475  do i = 1, offset
476  ac(:,i) = weights * dk(:,i)
477  end do
478  do i = 1, offset
479  ac(:,offset+i) = -weights * dk(:,i) * f
480  end do
481  ar(1:ns,:) = real(ac)
482  ar(ns+1:2*ns,:) = aimag(ac)
483 
484  ! Integral criteria for sigma
485  do i = 1, offset
486  ar(2*ns+1,offset+i) = real(scale * sum(dk(:,i)))
487  end do
488 
489  ! Compute the QR factorization of AR, and then form Q & R fully
490  call qr_factor(ar, tau)
491  call form_qr(ar, tau, q) ! R is stored in AR
492 
493  ! Extract the trailing submatrix from R
494  ind1 = offset + 1
495  ind2 = 2 * offset
496 
497  ! Construct B
498  b = q(size(q, 1), offset + 1:size(q, 2)) * scale * real(ns)
499 
500  ! Finish working on A
501  do i = 1, offset
502  ind = ind1 + i - 1
503  escale(i) = 1.0d0 / norm2(ar(:,ind))
504  ar(:,ind) = escale(i) * ar(:,ind)
505  end do
506 
507  ! Solve A * X = B, for X. Note: A is upper triangular
508  a = ar(ind1:ind2,ind1:ind2)
509  x = b ! Protect B from being overwritten
510  call solve_triangular_system(.true., .false., .true., a, x)
511 
512  ! Update X
513  x = x * escale
514 
515  ! Build C & D
516  c(1,:) = x(1:n)
517  d(1,1) = x(offset)
518 
519  ! Ensure D isn't too big or too small
520  if (abs(x(offset)) < tol_low .or. abs(x(offset)) > tol_high) then
521  if (x(offset) == 0.0d0) then
522  dnew = 1.0d0
523  else if (abs(x(offset)) < tol_low) then
524  dnew = sign(1.0d0, x(offset)) * tol_low
525  else if (abs(x(offset)) > tol_high) then
526  dnew = sign(1.0d0, x(offset)) * tol_high
527  end if
528 
529  do i = 1, offset
530  ac(:,i) = weights * dk(:,i)
531  end do
532  do i = 1, offset
533  ac(:,offset+i) = -weights * dk(:,i) * f
534  end do
535  ar(1:ns,:) = real(ac)
536  ar(ns+1:2*ns,:) = aimag(ac)
537  bc = dnew * weights * f
538  br(1:ns) = real(bc)
539  br(ns+1:2*ns) = aimag(bc)
540 
541  ! Compute the QR factorization of AR, and then form Q & R fully
542  call qr_factor(ar, tau)
543  call form_qr(ar, tau, q) ! R is stored in AR
544 
545  ! Extract the trailing submatrix from R, and store in the state-space
546  ! matrix A
547  ind1 = offset + 1
548  ind2 = 2 * offset
549  a = ar(ind1:ind2,ind1:ind2)
550 
551  ! Compute B = Q(:,IND1:IND2)**T * BR
552  !
553  ! - Q(:,IND1:IND2) is 2*NS-by-N
554  ! - BR is 2*NS-by-1
555  ! - B is then N-by-1
556  call dgemm('T', 'N', n, 1, 2*ns, 1.0d0, q, 2*ns, br, 2*ns, 0.0d0, b, n)
557 
558  ! Finish working on A
559  do i = 1, offset
560  escale(i) = 1.0d0 / norm2(a(:,i))
561  a(:,i) = escale(i) * a(:,i)
562  end do
563 
564  ! Solve A * X = B, for X. Note: A is upper triangular
565  x = b ! Protect B from being overwritten
566  call solve_triangular_system(.true., .false., .true., a, x)
567 
568  ! Update X
569  x = x * escale
570 
571  ! Build C & D
572  c(1,:) = x(1:n)
573  d(1,1) = dnew
574  end if
575  end subroutine

585  subroutine c_to_cmplx(c, cindex, cc)

592  complex(real64), parameter :: j = (0.0d0, 1.0d0)

606  cc(i+1) = r1 - j * r2

615 ! ------------------------------------------------------------------------------
616  ! Compute the zeros of the system.
617  !
618  ! - poles: On input, the current estimate of the pole locations. On output,
619  ! the updated estimate.
620  ! - b: An N-element complex-valued workspace array.
621  ! - c: An N-element complex valued "C" matrix from the state-space definition
622  ! of the system - the C matrix is 1-by-n.
623  ! - d: A 1-by-1 matrix containing the "D" matrix from the state-space
624  ! definition of the system.
625  ! - lambda: An N-by-N complex-valued workspace array.
626  subroutine compute_zeros(poles, b, c, d, lambda)
627  use linalg_core
628 
629  ! Arguments
630  complex(real64), intent(inout), dimension(:) :: poles
631  complex(real64), intent(out), dimension(:) :: b
632  complex(real64), intent(out), dimension(:,:) :: lambda
633  complex(real64), intent(inout), dimension(:) :: c
634  real(real64), intent(in), dimension(:,:) :: d
635 
636  ! Parameters
637  complex(real64), parameter :: zero = (0.0d0, 0.0d0)
638  complex(real64), parameter :: one = (1.0d0, 0.0d0)
639  complex(real64), parameter :: two = (2.0d0, 0.0d0)
640  complex(real64), parameter :: j = (0.0d0, 1.0d0)
641 
642  ! Local Variables
643  integer(int32) :: i, n, m, n1
644  complex(real64) :: koko, alpha, temp
645 
646  ! Initialization
647  n = size(poles)
648 
649  lambda = zero
650  do i = 1, n
651  b(i) = one
652  lambda(i,i) = poles(i)
653  end do
654 
655  ! Process
656  m = 0
657  do i = 1, n
658  m = m + 1
659  if (m < n) then
660  if (abs(lambda(m,m)) > abs(real(lambda(m,m)))) then
661  lambda(m+1,m) = -aimag(lambda(m,m))
662  lambda(m,m+1) = aimag(lambda(m,m))
663  lambda(m,m) = real(lambda(m,m))
664  lambda(m+1,m+1) = lambda(m,m)
665 
666  b(m) = two
667  b(m+1) = zero
668  koko = c(m)
669  c(m) = cmplx(real(koko), kind = real64)
670  c(m+1) = cmplx(aimag(koko), kind = real64)
671  m = m + 1
672  end if
673  end if
674  end do
675 
676  ! Construct the eigenvalue problem
677  ! ZER = LAMBDA - B * C / D(1,1) - use ZGEMM to compute
678  !
679  ! LAMBDA is N-by-N
680  ! B is N-by-1
681  ! C is 1-by-N
682  alpha = cmplx(-1.0d0 / d(1,1), kind = real64)
683  call zgemm('N', 'N', n, n, 1, alpha, b, n, c, 1, one, lambda, n)
684  ! Results of ZGEMM stored in LAMBDA
685 
686  ! Compute the eigenvalues
687  call eigen(lambda, poles)
688 
689  ! Correct any unstable poles - force them to be stable
690  ! (i.e. ensure the real components are negative)
691  do i = 1, n
692  if (real(poles(i)) > 0.0d0) then
693  ! The real component is positive - unstable pole
694  poles(i) = poles(i) - 2.0d0 * poles(i)
695  end if
696  end do
697 
698  ! Sort the poles into ascending order
699  call sort(poles, .true.)
700 
701  ! Now ensure real-valued poles are placed first in the array
702  do i = 1, n
703  do m = i + 1, n
704  if (aimag(poles(m)) == 0.0d0 .and. aimag(poles(i)) /= 0.0d0) then
705  temp = poles(i)
706  poles(i) = poles(m)
707  poles(m) = temp
708  end if
709  end do
710  end do
711  n1 = 0
712  do i = 1, n
713  if (aimag(poles(i)) == 0.0d0) n1 = i
714  end do
715  if (n1 < n) call sort(poles(n+1:n), .true.)
716  poles = poles - 2.0d0 * j * aimag(poles)
717  end subroutine

720  ! Constructs the C matrix into its complex-valued form.
721  !
722  ! - weights: M-element weighting vector.
723  ! - s: M-element array containing the frequency points.
724  ! - f: M-element frequency response vector containing data to be fitted.
725  ! - poles: An N-element array containing the current pole locations.

732  ! - bc: An M element array.
733  ! - br: A 2*M element array.
734  ! - cr: A 1-by-N matrix.
735  ! - dr: A 1-by-1 matrix.
736  ! - escale: An N+1 element array.
737  subroutine to_cmplx_state_space(weights, s, f, poles, cindex, dk, ac, ar, bc, br, cr, dr, escale)
738  use linalg_core
739 
740  ! Arguments
741  real(real64), intent(in), dimension(:) :: weights
742  complex(real64), intent(in), dimension(:) :: s, f, poles
743  integer(int32), intent(in), dimension(:) :: cindex
744  complex(real64), intent(inout), dimension(:,:) :: dk
745  complex(real64), intent(out), dimension(:,:) :: ac
746  complex(real64), intent(out), dimension(:) :: bc
747  real(real64), intent(out), dimension(:,:) :: ar, cr, dr
748  real(real64), intent(out), dimension(:) :: escale, br
749 
750  ! Parameters
751  complex(real64), parameter :: one = (1.0d0, 0.0d0)
752  complex(real64), parameter :: j = (0.0d0, 1.0d0)
753 
754  ! Local Variables
755  integer(int32) :: i, ns, n
756 
757  ! Initialization
758  ns = size(dk, 1)
759  n = size(dk, 2) - 1 ! DK is Ns -by- (N + 1)
760 
761  ! Process
762  i = 1
763  do while (i <= n)
764  if (cindex(i) == 0) then
765  dk(:,i) = weights / (s - poles(i))
766  i = i + 1
767  else if (cindex(i) == 1) then
768  dk(:,i) = weights / (s - poles(i)) + weights / (s - conjg(poles(i)))
769  dk(:,i+1) = j * weights / (s - poles(i)) - j * weights / (s - conjg(poles(i)))
770  i = i + 2
771  end if
772  end do
773 
774  do i = 1, n
775  ac(:,i) = weights * dk(:,i)
776  end do
777  ac(:,n+1) = cmplx(weights, kind = real64)
778  ar(ns+1:2*ns,:) = aimag(ac(1:ns,:))
779  ar(1:ns,:) = real(ac(1:ns,:))
780  bc(1:ns) = weights * f
781  br(ns+1:2*ns) = aimag(bc(1:ns))
782  br(1:ns) = real(bc(1:ns))
783 
784  ! Rescale A
785  do i = 1, n + 1
786  escale(i) = norm2(ar(:,i))
787  ar(:,i) = ar(:,i) / escale(i)
788  end do
789 
790  ! Solve A * X = B for X via a least-squares solver
791  call solve_least_squares(ar, br) ! AR & BR are overwritten
792  br(1:n+1) = br(1:n+1) / escale ! Rescale the solution
793  cr(1,:) = br(1:n) ! The first N elements of BR contain the solution for C
794 
795  ! The N+1th element of BR contains D
796  dr(1,1) = br(n+1)
797  end subroutine
