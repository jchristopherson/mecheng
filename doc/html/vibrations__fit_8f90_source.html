259 ! ------------------------------------------------------------------------------
260  ! Applies a relaxed vector fitting algorithm to the problem of fitting a
261  ! frequency response function of unknown origin. The code is a slightly
262  ! modified version of the MATLAB code by Bjorn Gustavsen available at
263  ! http://www.energy.sintef.no/Produkt/VECTFIT/index.asp.
264  !
265  ! - s: An N-element array containing the frequency values (rad/s).
266  ! - frf: An N-element array containing the frequency response to fit.
267  ! - poles: An M-element array containing the estimate of the system poles.
268  ! On output, this is updated to a new estimate of pole locations.
269  ! - weights: An N-element weighting vector that corresponds to each
270  ! input frequency.
271  ! - relax: Set to true to allow for relaxation; else, false.
272  ! - stable: Set to true to stabilize any unstable poles; else, false to
273  ! allow unstable poles.
274  ! - mdl [Output]: A state-space model where the matrices are dimensioned
275  ! as follows:
276  ! - A: M-by-M
277  ! - B: M-by-1
278  ! - C: 1-by-M
279  ! - D: 1-by-1
280  ! - fit [Output]: An N-element array containing the fitted data.
281  ! - dif [Output]: An N-element array containing the difference between
282  ! fitted and actual data.
283  ! - work: A K-element workspace array where K is 2*M**2 + (8*N + 10)*M + 10*N + 9.
284  ! - cwork: An L-element workspace array where L is (3*N + 2)*M + 4*N.
285  ! - iwork: An M-element workspace array.
286  ! - err: An error handler used to capture any errors from QR factorization
287  ! or eigenvalue extraction routines.
288  !
289  ! References:
290  ! - B. Gustavsen and A. Semlyen, "Rational approximation of frequency
291  ! domain responses by Vector Fitting", IEEE Trans. Power Delivery,
292  ! vol. 14, no. 3, pp. 1052-1061, July 1999.
293  ! - B. Gustavsen, "Improving the pole relocating properties of vector
294  ! fitting", IEEE Trans. Power Delivery, vol. 21, no. 3, pp. 1587-1592,
295  ! July 2006.
296  ! - D. Deschrijver, M. Mrozowski, T. Dhaene, and D. De Zutter,
297  ! "Macromodeling of Multiport Systems Using a Fast Implementation of
298  ! the Vector Fitting Method", IEEE Microwave and Wireless Components
299  ! Letters, vol. 18, no. 6, pp. 383-385, June 2008.
300  subroutine frf_fit_core(s, frf, poles, weights, relax, stable, mdl, &
301  fit, dif, work, cwork, iwork, err)
302  use linalg_core
303 
304  ! Arguments
305  complex(real64), intent(in), dimension(:) :: s, frf
306  complex(real64), intent(inout), dimension(:) :: poles
307  real(real64), intent(in), dimension(:) :: weights
308  logical, intent(in) :: relax, stable
309  class(state_space), intent(inout) :: mdl
310  real(real64), intent(out), target, dimension(:) :: work
311  complex(real64), intent(out), target, dimension(:) :: fit, dif, cwork
312  integer(int32), intent(out), target, dimension(:) :: iwork
313  class(errors), intent(inout) :: err
314 
315  ! Parameters
316  complex(real64), parameter :: zero = (0.0d0, 0.0d0)
317  complex(real64), parameter :: one = (1.0d0, 0.0d0)
318  complex(real64), parameter :: i1 = (0.0d0, 1.0d0)
319  real(real64), parameter :: tolLow = 1.0d-18
320  real(real64), parameter :: tolHi = 1.0d18
321 
322  ! Local Variables
323  integer(int32) :: i, j, m, n, np, np1, ind1, ind2, &
324  e1, s2, e2, s3, e3, s4, e4, s5, e5, s6, e6, s7, e7, s8, e8, &
325  e1c, s2c, e2c, s3c, e3c, s4c, e4c, s5c, e5c
326  real(real64) :: scale, Dnew, r1, r2
327  integer(int32), pointer, dimension(:) :: cindex
328  real(real64), pointer, dimension(:) :: tau, x, Escale, b
329  real(real64), pointer, dimension(:,:) :: A, Q, R, lambda
330  complex(real64) :: pole
331  complex(real64), pointer, dimension(:) :: bc, Cc, eigenvals
332  complex(real64), pointer, dimension(:,:) :: Dk, Ac
333 
334  ! Initialization
335  n = size(s)
336  np = size(poles)
337  np1 = np + 1
338 
339  ! Define limits within the workspace arrays
340  e1 = (2 * n + 1) * 2 * np1
341  s2 = e1 + 1
342  e2 = s2 + 2 * np1 - 1
343  s3 = e2 + 1
344  e3 = s3 + (2 * n + 1) * 2 * np1 - 1
345  s4 = e3 + 1
346  e4 = s4 + np1 * np1 - 1
347  s5 = e4 + 1
348  e5 = s5 + np
349  s6 = e5 + 1
350  e6 = s6 + np
351  s7 = e6 + 1
352  e7 = s7 + 2 * n - 1
353  s8 = e7 + 1
354  e8 = s8 + np * np - 1
355 
356  e1c = n * np1
357  s2c = e1c + 1
358  e2c = s2c + n * (2* np1) - 1
359  s3c = e2c + 1
360  e3c = s3c + n - 1
361  s4c = e3c + 1
362  e4c = s4c + np - 1
363  s5c = e4c + 1
364  e5c = s5c + np - 1
365 
366  ! Assign Pointers
367  cindex => iwork(1:np) ! NP-element
368 
369  a(1:2*n+1,1:2*np1) => work(1:e1) ! 2*N+1-by-2*(NP+1)
370  tau => work(s2:e2) ! 2*(NP+1)-element
371  q(1:2*n+1,1:2*np1) => work(s3:e3) ! 2*N+1-by-2*(NP+1)
372  r(1:np1,1:np1) => work(s4:e4) ! (NP+1)-by-(NP+1)
373  x => work(s5:e5) ! (NP+1)-element
374  escale => work(s6:e6) ! (NP+1)-element
375  b => work(s7:e7) ! 2*N-element
376  lambda(1:np,1:np) => work(s8:e8) ! NP-by-NP
377 
378  dk(1:n,1:np1) => cwork(1:e1c) ! N-by-(NP+1)
379  ac(1:n,1:2*np1) => cwork(s2c:e2c) ! N-by-2*(NP+1)
380  bc => cwork(s3c:e3c) ! N-element
381  cc => cwork(s4c:e4c) ! NP-element
382  eigenvals => cwork(s5c:e5c) ! NP-element
383 
384  ! Locate the complex poles
385  call find_complex(poles, cindex)
386 
387  ! Build the system matrices
388  do i = 1, np
389  if (cindex(i) == 0) then
390  dk(:,i) = one / (s - poles(i))
391  else if (cindex(i) == 1) then
392  dk(:,i) = one / (s - poles(i)) + one / (s - conjg(poles(i)))
393  dk(:,i+1) = i1 / (s - poles(i)) - i1 / (s - conjg(poles(i)))
394  end if
395  end do
396  dk(:,np1) = one
397  scale = norm2(abs(weights * frf)) / real(n)
398 
399  ! **********************************************************************
400  if (relax) then
401  do i = 1, np1
402  ac(:,i) = weights * dk(:,i)
403  ac(:,i+np1) = -weights * dk(:,i) * frf
404  end do
405  a(1:n,:) = real(Ac)
406  a(n+1:2*n,:) = aimag(ac)
407 
408  ! Establish the criteria for sigma
409  a(2*n+1,1:np1) = 0.0d0
410  do i = 1, np1
411  a(2*n+1,np1+i) = real(scale * sum(Dk(:,i)))
412  end do
413 
414  ! Compute the QR factorization - economy version of Q.
415  ! R overwrites A, and it is upper triangular
416  call qr_factor(a, tau, err = err)
417  if (err%has_error_occurred()) return
418 
419  ! Form the economy version of Q [2*N+1 -by- 2*(NP+1)]
420  call form_qr(a, tau, q, err = err)
421  if (err%has_error_occurred()) return
422 
423  ind1 = np + 2
424  ind2 = 2 * np1
425  r = a(ind1:ind2,ind1:ind2)
426  x = scale * real(n) * Q(size(Q, 1), ind1:ind2)
427 
428  do i = 1, np1
429  escale(i) = 1.0d0 / norm2(r(:,i))
430  r(:,i) = escale(i) * r(:,i)
431  end do
432 
433  ! Solve the upper triangular system
434  call solve_triangular_system(.true., .false., .true., r, x, err)
435  if (err%has_error_occurred()) return
436 
437  ! Recover the scaling
438  x = x * escale
439  end if
440 
441  ! Deal with the situation when no relaxation is requested, or D
442  ! is too small
443  if (.not.relax .or. abs(x(np1)) < tollow .or. abs(x(np1)) > tolhi) then
444  if (relax) then
445  dnew = 1.0d0
446  else
447  if (x(np1) == 0.0d0) then
448  dnew = 1.0d0
449  else if (abs(x(np1)) < tollow) then
450  dnew = sign(tollow, x(np1))
451  else if (abs(x(np1)) > tolhi) then
452  dnew = sign(tolhi, x(np1))
453  end if
454  end if
455 
456  do i = 1, np1
457  ac(:,i) = weights * dk(:,i)
458  end do
459  do i = 1, np
460  ac(:,i+np1) = -weights * dk(:,i) * frf
461  end do
462  bc = dnew * weights * frf
463 
464  a(1:n,1:2*np+1) = real(Ac(:,1:2*np+1))
465  a(n+1:2*n,1:2*np+1) = aimag(ac(:,1:2*np+1))
466  b(1:n) = real(bc)
467  b(n+1:2*n) = aimag(bc)
468 
469  ! Compute the QR factorization of the 2*N-by-2*NP+1 matrix A
470  call qr_factor(a(1:2*n,1:2*np+1), tau(1:2*np+1), err = err)
471  if (err%has_error_occurred()) return
472 
473  ! Form the economy version of Q [2*N -by- 2*(NP+1)]
474  call form_qr(a(1:2*n,1:2*np+1), tau(1:2*np+1), q, err = err)
475  if (err%has_error_occurred()) return
476 
477  ind1 = np + 2
478  ind2 = 2 * np + 1
479  r = a(ind1:ind2,ind1:ind2) ! R is NP-by-NP
480 
481  ! Compute X = Q**T * B where Q**T is NP-by-2*N & b is 2*N such that x is NP
482  call dgemv('T', 2*n, np, 1.0d0, q(1:2*n,ind1:ind2), 2*n, b, 1, 0.0d0, x(1:np), 1)
483 
484  ! Scale the problem
485  do i = 1, np
486  escale(i) = 1.0d0 / norm2(r(:,i))
487  r(:,i) = escale(i) * r(:,i)
488  end do
489 
490  ! Solve the upper-triangular problem
491  call solve_triangular_system(.true., .false., .true., r(1:np,1:np), x(1:np), err)
492  if (err%has_error_occurred()) return
493 
494  ! Recover the scaling
495  x(1:np) = x(1:np) * escale(1:np)
496 
497  ! Store Dnew
498  x(np1) = dnew
499  end if
500 
501  ! Now, generate an estimate for C and D
502  mdl%C(1,1:np) = x(1:np)
503  mdl%D(1,1) = x(np1)
504 
505  ! **********************************************************************
506  ! Make C complex-valued
507  i = 1
508  do while (i <= np)
509  if (cindex(i) == 1) then
510  r1 = mdl%C(1,i)
511  r2 = mdl%C(1,i+1)
512  cc(i) = r1 + i1 * r2
513  cc(i+1) = r1 - i1 * r2
514  i = i + 2
515  else
516  cc(i) = zero
517  i = i + 1
518  end if
519  end do
520 
521  ! **********************************************************************
522  ! Compute the zeros
523  lambda = 0.0d0
524  m = 0
525  mdl%B = 0.0d0
526  mdl%C = 0.0d0
527  do i = 1, np
528  m = m + 1
529  if (m < np) then
530  pole = poles(m)
531  if (abs(pole) > abs(real(pole))) then
532  ! The pole is complex-valued
533  lambda(m+1,m) = -aimag(pole)
534  lambda(m,m+1) = aimag(pole)
535  lambda(m,m) = real(pole)
536  lambda(m+1,m+1) = real(pole)
537 
538  mdl%B(m,1) = 2.0d0
539  mdl%B(m+1,1) = 0.0d0
540 
541  mdl%C(1,m) = real(Cc(m))
542  mdl%C(1,m+1) = aimag(cc(m))
543 
544  m = m + 1
545  end if
546  end if
547  end do
548 
549  ! Compute ZER = LAMBDA - B * C / D - use LAMBDA to store ZER
550  call dgemm('N', 'N', np, np, 1, -1.0d0 / mdl%D(1,1), mdl%B, np, &
551  mdl%C, 1, 1.0d0, lambda, np)
552 
553  ! Compute the eigenvalues - lambda is overwritten by this call
554  call eigen(lambda, eigenvals, err = err)
555  if (err%has_error_occurred()) return
556 
557  ! Force stability?
558  if (stable) then
559  do i = 1, np
560  if (real(eigenvals(i)) > 0.0d0) then
561  eigenvals(i) = eigenvals(i) - 2.0d0 * real(eigenvals(i))
562  end if
563  end do
564  end if
565 
566  ! Sort the eigenvalus in ascending order
567  call sort(eigenvals, .true.)
568 
569  ! Move any real poles to the beginning of the eigenvalue list
570  do i = 1, np
571  do j = i+1, np
572  if (aimag(eigenvals(j)) == 0.0d0 .and. aimag(eigenvals(i)) /= 0.0d0) then
573  pole = eigenvals(i);
574  eigenvals(i) = eigenvals(j)
575  eigenvals(j) = pole
576  end if
577  end do
578  end do
579  m = 0
580  do i = 1, np
581  if (aimag(eigenvals(i)) == 0.0d0) then
582  m = i
583  end if
584  end do
585  if (m < np) then
586  call sort(eigenvals(m+1:np), .true.)
587  end if
588  eigenvals = eigenvals - 2.0d0 * i1 * aimag(eigenvals)
589 
590  ! **********************************************************************
591  ! Residual Identification
592  poles = eigenvals
593  call find_complex(poles, cindex)
594 
595  ! Compute the new fitting using the new poles estimate
596  do i = 1, np
597  if (cindex(i) == 0.0d0) then
598  dk(:,i) = weights / (s - poles(i))
599  else if (cindex(i) == 1) then
600  dk(:,i) = weights * (one / (s - poles(i)) + one / (s - conjg(poles(i))))
601  dk(:,i+1) = i1 * weights * (one / (s - poles(i)) - one / (s - conjg(poles(i))))
602  end if
603  end do
604 
605  ac(:,1:np) = dk(:,1:np)
606  ac(:,np1) = weights
607  bc = weights * frf
608  a(1:n,1:np1) = real(Ac(:,1:np1))
609  a(n+1:2*n,1:np1) = aimag(ac(:,1:np1))
610  b(1:n) = real(bc)
611  b(n+1:2*n) = aimag(bc)
612 
613  ! Scale the matrices
614  do i = 1, np1
615  escale(i) = norm2(a(:,i))
616  a(:,i) = a(:,i) / escale(i)
617  end do
618 
619  ! Solve the least-squares problem
620  ! A is 2*N-by-NP+1 & b is 2*N, so X is NP+1
621  call solve_least_squares(a(1:2*n,1:np1), b, err = err)
622  if (err%has_error_occurred()) return
623 
624  ! Rescale the solution
625  x = b(1:np1) / escale
626 
627  ! Redefine C & D
628  mdl%C(1,1:np) = x(1:np)
629  mdl%D(1,1) = x(np1)
630 
631  ! Make C complex-valued
632  i = 1
633  do while (i <= np)
634  if (cindex(i) == 1) then
635  r1 = mdl%C(1,i)
636  r2 = mdl%C(1,i+1)
637  cc(i) = r1 + i1 * r2
638  cc(i+1) = r1 - i1 * r2
639  i = i + 2
640  else
641  cc(i) = zero
642  i = i + 1
643  end if
644  end do
645 
646  ! **********************************************************************
647  ! Compute the fit, and then determine the error
648  do i = 1, np
649  dk(:,i) = one / (s - poles(i))
650  end do
651 
652  ! Compute FIT = Dk(:,1:np) * Cc + D
653  fit = mdl%D(1,1)
654  call zgemv('N', n, np, one, dk(:,1:np), n, cc, 1, one, fit, 1)
655 
656  ! Compute the error
657  dif = fit - frf
658 
659  ! **********************************************************************
660  ! Convert into a real-valued state-space model
661  j = 0
662  mdl%A = 0.0d0
663  mdl%B = 1.0d0
664  do i = 1, np
665  j = j + 1
666  if (cindex(i) == 1) then
667  mdl%A(j,j) = real(poles(j))
668  mdl%A(j+1,j) = -aimag(poles(j))
669  mdl%A(j,j+1) = aimag(poles(j))
670  mdl%A(j+1,j+1) = mdl%A(j,j)
671 
672  mdl%B(j,1) = 2.0d0 * real(mdl%B(j,1))
673  mdl%B(j+1,1) = 0.0d0
674  ! mdl%B(j+1,1) = -2.0d0 * aimag(mdl%B(j,1))
675 
676  mdl%C(1,j) = real(Cc(j))
677  mdl%C(1,j+1) = aimag(cc(j))
678  end if
679  end do
680  end subroutine
