10  type(frequency_sweep_options), intent(in), optional :: opt
11  class(errors), intent(inout), optional, target :: err
12  real(real64), allocatable, dimension(:,:) :: rst
13 
14  ! Local Variables
15  logical :: status
16  integer(int32) :: neqn, nfreq, ncycle, scycle, flag
17  real(real64), allocatable, dimension(:) :: ic, w1, w2
18  class(errors), pointer :: errmgr
19  type(errors), target :: deferr

22 
23  ! Initialization
24  if (present(err)) then
25  errmgr => err
26  else
27  errmgr => deferr
28  end if
29  neqn = size(xo)
30  nfreq = size(freq)
31 
32  ! Establish settings
33  if (present(opt)) then
34  integrator => opt%integrator
35  ncycle = opt%forced_cycle_count
36  scycle = opt%measured_cycle_count
37  status = opt%display_status
38  else
39  integrator => defintegrator
40  ncycle = 50
41  scycle = 10
42  status = .false.
43  end if
44 
45  ! Verify the settings
46  if (.not.associated(integrator)) then
47  call errmgr%report_error("compute_frequency_sweep", &
48  "No itegrator is defined.", mech_null_reference_error)
49  return
50  end if
51  if (ncycle < 2) then
52  call errmgr%report_error("compute_frequency_sweep", &
53  "At least 2 forcing cycles must be employed.", &
54  mech_invalid_input_error)
55  return
56  end if
57  if (scycle >= ncycle) then
58  call errmgr%report_error("compute_frequency_sweep", &
59  "The number of sampling cycles cannot match or exceed " // &
60  "the number of forcing cycles.", &
61  mech_invalid_input_error)
62  return
63  end if
64  if (scycle < 1) then
65  call errmgr%report_error("compute_frequency_sweep", &
66  "At least 1 sample cycle must be employed.", &
67  mech_invalid_input_error)
68  return
69  end if
70 
71  ! Local Memory Allocation
72  allocate(ic(neqn), stat = flag)
73  if (flag == 0) allocate(w1(neqn), stat = flag)
74  if (flag == 0) allocate(w2(neqn), stat = flag)
75  if (flag == 0) allocate(rst(nfreq, neqn), stat = flag)
76  if (flag /= 0) then
77  call errmgr%report_error("compute_frequency_sweep", &
78  "Insufficient memory available.", &
79  mech_out_of_memory_error)

81  end if

85  call sweep_engine(integrator, fcn, freq, ic, ncycle, scycle, &

89 ! ------------------------------------------------------------------------------
90  ! Sweeps through the supplied frequency vector, and computes the amplitude
91  ! of the resulting outputs.
92  !
93  ! Inputs:
94  ! - integrator: The integrator to use. The reset routine of the integrator
95  ! is utilized between iterations to ensure the integrator is properly
96  ! aligned for each iteration.
97  ! - fcn: A pointer to the routine containing the ODE's to integrate.
98  ! - freq: An N-element array containing the frequency points, in rad/s.
99  ! - xo: An M-element array containing the initial conditions for each of
100  ! the M ODE's.
101  ! - ncycle: The number of cycles of forced oscillations the system of ODE's
102  ! is forced through.
103  ! - scycle: The number of forced oscillation cycles at the end of the signal
104  ! that are used to determine the output.
105  ! - status: Set to true to update status on each iteration; else, false for
106  ! no updates.
107  ! - err: The error handler object.
108  !
109  ! Output:
110  ! - xo: The M-element initial condition array is used as a working array to
111  ! update initial conditions for each solution step.
112  ! - amp: An N-by-M matrix containing the responses from each of the M equations.
113  ! - work1: An M-element workspace array.
114  ! - work2: An M-element workspace array.
115  subroutine sweep_engine(integrator, fcn, freq, xo, ncycle, scycle, status, &
116  amp, work1, work2, err)
117  use constants
118 
119  ! Arguments
120  class(ode_integrator), intent(inout) :: integrator
121  procedure(harmonic_ode_fcn), intent(in), pointer :: fcn
122  real(real64), intent(in), dimension(:) :: freq
123  real(real64), intent(inout), dimension(:) :: xo
124  integer(int32), intent(in) :: ncycle, scycle
125  logical, intent(in) :: status
126  real(real64), intent(out), dimension(:,:) :: amp
127  real(real64), intent(out), dimension(:) :: work1, work2
128  class(errors), intent(inout) :: err
129 
130  ! Local Variables
131  type(ode_helper) :: ode
132  procedure(ode_fcn), pointer :: fcnptr
133  integer(int32) :: i, nfreq, neqn, npts, index
134  real(real64) :: currentFrequency, tstart, time(2)
135  real(real64), allocatable, dimension(:,:) :: rst
136 
137  ! Initialization
138  nfreq = size(freq)
139  neqn = size(xo)
140 
141  ! Set up the integrator
142  fcnptr => eom
143  call ode%define_equations(neqn, fcnptr)
144 
145  ! Cycle over each frequency. This must be done in a serial manner
146  ! as each solution segment for a given frequency relies upon the
147  ! solution of the prior step to establish its initial conditions.
148  do i = 1, nfreq
149  ! Establish the current frequency
150  currentfrequency = freq(i)
151 
152  ! Reset the integrator such that it is good to go for this round
153  call integrator%reset()
154 
155  ! Establish the solution time vector
156  time = [0.0d0, 2.0d0 * pi * ncycle / currentfrequency]
157 
158  ! Perform the integration
159  rst = integrator%integrate(ode, time, xo, err)
160  if (err%has_error_occurred()) return ! Exit if an error was encountered
161 
162  ! Determine the region to search for max/min values
163  tstart = 2.0d0 * pi * (ncycle - scycle) / currentfrequency
164  index = find(rst(:,1), tstart)
165 
166  ! Search over the desired range for the max and min values
167  npts = size(rst, 1)
168  call find_extremes(rst(index:npts, 2:neqn+1), work1, work2)
169 
170  ! Compute the amplitude
171  amp(i,:) = 0.5d0 * (work1 - work2)
172 
173  ! Establish the new set of initial conditions
174  xo = rst(npts, 2:neqn+1)
175 
176  ! Display a status update?
177  if (status) then
178  print '(AI0AI0AES11.4A)', "Iteration: ", i, " of ", nfreq, &
179  ", Frequency: ", currentfrequency, " rad/s."
180  end if
181  end do

198  ! Finds the index of the first value in x to be >= to lim.
199  pure function find(x, lim) result(ind)
200  ! Arguments
201  real(real64), intent(in), dimension(:) :: x
202  real(real64), intent(in) :: lim
203  integer(int32) :: ind
204

206  integer(int32) :: i, n
207 
208  ! Process

215  end if

219 ! ------------------------------------------------------------------------------
220  ! Finds the maximum and minimum values in each column of x.
221  subroutine find_extremes(x, big, small)
222  ! Arguments
223  real(real64), intent(in), dimension(:,:) :: x
224  real(real64), intent(out), dimension(:) :: big, small
225 
226  ! Process
227  integer(int32) :: i, j, m, n
228 
229  ! Initialization
230  m = size(x, 1)
231  n = size(x, 2)
232 
233  ! Quick Return
234  if (m == 0 .or. n == 0) return
235 
236  ! Process
237  do j = 1, n
238  big(j) = x(1,j)
239  small(j) = x(1,j)
240  end do
241  do j = 1, n
242  do i = 2, m
243  if (x(i,j) > big(j)) big(j) = x(i,j)
244  if (x(i,j) < small(j)) small(j) = x(i,j)
245  end do
246  end do
247  end subroutine
