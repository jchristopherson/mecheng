277  subroutine er_report_error(this, fcn, msg, flag, obj)
278  ! Arguments
279  class(errors), intent(inout) :: this
280  character(len = *), intent(in) :: fcn, msg
281  integer(int32), intent(in) :: flag
282  class(*), intent(inout), optional :: obj
283 
284  ! Local Variables
285  integer(int32) :: n, dummy
286 
287  ! Write the error message to the command line
288  if (.not.this%m_suppressPrinting) then
289  print *, ""
290  print '(A)', "***** ERROR *****"
291  print '(A)', "Function: " // fcn
292  print '(AI0)', "Error Flag: ", flag
293  print '(A)', "Message:"
294  print '(A)', msg
295  print *, ""
296  end if
297 
298  ! Update the error found status
299  this%m_foundError = .true.
300  this%m_errorFlag = flag
301 
302  ! Store the message
303  n = len(msg)
304  if (allocated(this%m_errorMessage)) deallocate(this%m_errorMessage)
305  allocate(character(len = n) :: this%m_errorMessage)
306  this%m_errorMessage = msg(1:n)
307 
308  ! Store the function name
309  n = len(fcn)
310  if (allocated(this%m_eFunName)) deallocate(this%m_eFunName)
311  allocate(character(len = n) :: this%m_eFunName)
312  this%m_eFunName = fcn(1:n)
313 
314  ! Write the error message to a log file
315  call this%log_error(fcn, msg, flag)
316 
317  ! Call the clean-up routine, if available
318  if (associated(this%m_errCleanUp)) then
319  if (present(obj)) then
320  call this%m_errCleanUp(obj)
321  else
322  dummy = 0
323  call this%m_errCleanUp(dummy)
324  end if
325  end if
326 
327  ! Exit the program
328  if (this%m_exitOnError) call exit(flag)
329  end subroutine
