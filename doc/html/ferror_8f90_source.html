282  subroutine er_report_error(this, fcn, msg, flag, obj)
283  ! Arguments
284  class(errors), intent(inout) :: this
285  character(len = *), intent(in) :: fcn, msg
286  integer(int32), intent(in) :: flag
287  class(*), intent(inout), optional :: obj
288 
289  ! Local Variables
290  integer(int32) :: n, dummy
291 
292  ! Write the error message to the command line
293  if (.not.this%m_suppressPrinting) then
294  print *, ""
295  print '(A)', "***** ERROR *****"
296  print '(A)', "Function: " // fcn
297  print '(AI0)', "Error Flag: ", flag
298  print '(A)', "Message:"
299  print '(A)', msg
300  print *, ""
301  end if
302 
303  ! Update the error found status
304  this%m_foundError = .true.
305  this%m_errorFlag = flag
306 
307  ! Store the message
308  n = len(msg)
309  if (allocated(this%m_errorMessage)) deallocate(this%m_errorMessage)
310  allocate(character(len = n) :: this%m_errorMessage)
311  this%m_errorMessage = msg(1:n)
312 
313  ! Store the function name
314  n = len(fcn)
315  if (allocated(this%m_eFunName)) deallocate(this%m_eFunName)
316  allocate(character(len = n) :: this%m_eFunName)
317  this%m_eFunName = fcn(1:n)
318 
319  ! Write the error message to a log file
320  call this%log_error(fcn, msg, flag)
321 
322  ! Call the clean-up routine, if available
323  if (associated(this%m_errCleanUp)) then
324  if (present(obj)) then
325  call this%m_errCleanUp(obj)
326  else
327  dummy = 0
328  call this%m_errCleanUp(dummy)
329  end if
330  end if
331 
332  ! Exit the program
333  if (this%m_exitOnError) call exit(flag)
334  end subroutine
