82  ! The transfer function H(s) can be computed as follows:
83  ! H(s) = C * (s * I - A)**-1 * B + D. The matrix inversion
84  ! component is more readily computed by letting
85  ! X = (s * I - A)**-1 * B (i.e. solving (s * I - A) * X = B)
86  ! by means of LU or QR factorization. As s * I - A should be
87  ! full rank, LU factorization with partial pivoting is the
88  ! preferred method.

93  if (flag == 0) allocate(h(nout, nin), stat = flag)
94  if (flag == 0) allocate(pvt(n), stat = flag)
95  if (flag == 0) allocate(c(nout, n), stat = flag)
96  if (flag /= 0) then

98  "Insufficient memory available.", &
99  mech_out_of_memory_error)
100  return

104  call compute_h(this%A, this%B, this%C, this%D, s, h, a, x, c, pvt)

172  ! Computes: H(s) = C * (s * I - A) * B + D
173  subroutine compute_h(A, B, C, D, s, H, workA, workB, workC, iwork)
174  use linalg_core
175 
176  ! Arguments
177  real(real64), intent(in), dimension(:,:) :: A, B, C, D
178  complex(real64), intent(in) :: s
179  complex(real64), intent(out), dimension(:,:) :: H, workA, workB, workC
180  integer(int32), intent(out), dimension(:) :: iwork
181 
182  ! Parameters
183  complex(real64), parameter :: one = (1.0d0, 0.0d0)

185  ! Local Variables

188  ! Initialization
189  n = size(a, 1)
190  nin = size(h, 1)
191  nout = size(h, 2)
192 
193  ! Process
194  do j = 1, n
195  do i = 1, n
196  if (i == j) then
197  worka(i,j) = s - a(i,j)
198  else
199  worka(i,j) = -cmplx(a(i,j), kind = real64)
200  end if
201  end do
202  end do
203 
204  ! Solve (s * I - A) * X = B for X
205  call lu_factor(worka, iwork)
206  workb = cmplx(b, kind = real64)
207  call solve_lu(worka, iwork, workb) ! workB holds X upon completion of this call
208 
209  ! Compute H = C * X + D as X = (s * I - A)**-1 * B
210  h = cmplx(d, kind = real64)
211  workc = cmplx(c, kind = real64)
212  call zgemm('N', 'N', nin, nout, n, one, workc, nin, workb, n, one, h, nin)
213  end subroutine
