104  subroutine sb_append(this, txt)
105  ! Arguments
106  class(string_builder), intent(inout) :: this
107  character(len = *), intent(in) :: txt
108 
109  ! Local Variables
110  integer(int32) :: space, n, start, finish, nb
111  character(len = :), allocatable :: temp
112 
113  ! Process
114  if (.not.allocated(this%m_buffer)) call this%initialize()
115  space = len(this%m_buffer) - this%m_length
116  n = len(txt)
117  if (space < n) then
118  ! Reallocate a larger buffer
119  nb = len(this%m_buffer)
120  allocate(character(len = nb + max(n, STRING_BUFFER_SIZE)) :: temp)
121  temp(1:nb) = this%m_buffer
122  this%m_buffer = temp
123  end if
124  start = this%m_length + 1
125  finish = start + n - 1
126  this%m_buffer(start:finish) = txt(1:n)
127  this%m_length = this%m_length + n
128  end subroutine

163  pure function split_string(txt, delim) result(rst)
164  ! Arguments
165  character(len = *), intent(in) :: txt
166  character, intent(in), optional :: delim
167  type(string), allocatable, dimension(:) :: rst
168 
169  ! Local Variables
170  integer(int32) :: i, j, k, n
171  character :: delimiter
172  character(len = len(txt)) :: buffer
173  type(string) :: stbuffer(len(txt))
174 
175  ! Initialization
176  n = len(txt)
177  if (present(delim)) then
178  delimiter = delim
179  else
180  delimiter = ","
181  end if
182 
183  ! Process
184  j = 0
185  k = 0
186  do i = 1, n
187  if (txt(i:i) /= delimiter) then
188  j = j + 1
189  buffer(j:j) = txt(i:i)
190  else
191  ! Reached a delimiter character
192  if (j == 0) cycle
193  k = k + 1
194  stbuffer(k)%str = buffer(1:j)
195  j = 0
196  end if
197  end do
198 
199  ! Catch the last string (between the final delimiter and end of txt)
200  if (j /= 0) then
201  k = k + 1
202  stbuffer(k)%str = buffer(1:j)
203  end if
204 
205  ! End
206  rst = stbuffer(1:k)
207  end function

354  pure function remove_substring(str, s) result(rst)
355  ! Arguments
356  character(len = *), intent(in) :: str, s
357  character(len = :), allocatable :: rst
358 
359  ! Local Variables
360  integer(int32) :: i, start, finish, nstr, ns, nb
361  character(len = len(str)) :: buffer
362 
363  ! Process

365  ns = len(s)

701 
702  ! Initialization

707  maxchar = 10000
708  end if
