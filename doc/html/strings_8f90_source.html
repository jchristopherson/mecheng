100  ! Arguments
101  class(string_builder), intent(inout) :: this
102  character(len = *), intent(in) :: txt
103 
104  ! Local Variables
105  integer(int32) :: space, n, start, finish, nb
106  character(len = :), allocatable :: temp
107 
108  ! Process
109  if (.not.allocated(this%m_buffer)) call this%initialize()
110  space = len(this%m_buffer) - this%m_length
111  n = len(txt)
112  if (space < n) then
113  ! Reallocate a larger buffer

116  temp(1:nb) = this%m_buffer
117  this%m_buffer = temp
118  end if
119  start = this%m_length + 1
120  finish = start + n - 1
121  this%m_buffer(start:finish) = txt(1:n)
122  this%m_length = this%m_length + n
123  end subroutine

158  pure function split_string(txt, delim) result(rst)
159  ! Arguments
160  character(len = *), intent(in) :: txt
161  character, intent(in), optional :: delim
162  type(string), allocatable, dimension(:) :: rst
163 
164  ! Local Variables
165  integer(int32) :: i, j, k, n
166  character :: delimiter
167  character(len = len(txt)) :: buffer
168  type(string) :: stbuffer(len(txt))
169 
170  ! Initialization
171  n = len(txt)
172  if (present(delim)) then
173  delimiter = delim
174  else
175  delimiter = ","
176  end if
177 
178  ! Process
179  j = 0
180  k = 0
181  do i = 1, n
182  if (txt(i:i) /= delimiter) then
183  j = j + 1
184  buffer(j:j) = txt(i:i)
185  else
186  ! Reached a delimiter character
187  if (j == 0) cycle
188  k = k + 1
189  stbuffer(k)%str = buffer(1:j)
190  j = 0
191  end if
192  end do
193 
194  ! Catch the last string (between the final delimiter and end of txt)
195  if (j /= 0) then
196  k = k + 1
197  stbuffer(k)%str = buffer(1:j)
198  end if
199 
200  ! End
201  rst = stbuffer(1:k)
202  end function

350  ! Arguments
351  character(len = *), intent(in) :: str, s
352  character(len = :), allocatable :: rst
353 
354  ! Local Variables
355  integer(int32) :: i, start, finish, nstr, ns, nb

357 
358  ! Process
