191  pure function split_string_pure(txt, delim, nbuff, buffsize) result(rst)
192  ! Arguments
193  character(len = *), intent(in) :: txt
194  character, intent(in), optional :: delim
195  integer(int32), intent(in), optional :: nbuff, buffsize
196  type(string), allocatable, dimension(:) :: rst
197 
198  ! Parameters
199  integer(int32), parameter :: default_line_size = 4096
200  integer(int32), parameter :: default_buffer_size = 10000
201 
202  ! Local Variables
203  integer(int32) :: i, j, k, n, buffersize, linesize
204  character :: delimiter
205  character(len = :), allocatable :: buffer, linecopy
206  type(string), allocatable, dimension(:) :: stbuffer, buffercopy
207 
208  ! Initialization
209  n = len(txt)
210  if (present(delim)) then
211  delimiter = delim
212  else
213  delimiter = ","
214  end if
215 
216  if (present(nbuff)) then
217  buffersize = nbuff
218  else
219  buffersize = default_buffer_size
220  end if
221  allocate(stbuffer(buffersize))
222 
223  if (present(buffsize)) then
224  linesize = buffsize
225  else
226  linesize = default_line_size
227  end if
228  allocate(character(len = lineSize) :: buffer)
229 
230 
231  ! Process
232  j = 0
233  k = 0
234  do i = 1, n
235  if (txt(i:i) /= delimiter) then
236  j = j + 1
237  if (j > linesize) then
238  ! Create a temporary copy of the line buffer
239  allocate(character(len = j - 1) :: linecopy)
240  linecopy = buffer
241 
242  ! Deallocate the line buffer, and then reallocate for a larger buffer
243  deallocate(buffer)
244  linesize = linesize + default_line_size
245  allocate(character(len = lineSize) :: buffer)
246 
247  ! Copy back the original buffer contents
248  buffer(1:j-1) = linecopy
249  deallocate(linecopy)
250  end if
251  buffer(j:j) = txt(i:i)
252  else
253  ! Reached a delimiter character
254  k = k + 1
255  if (k > buffersize) then
256  ! Create a temporary copy of the string buffer
257  allocate(buffercopy(k - 1))
258  buffercopy = stbuffer
259 
260  ! Deallocate the string buffer, and then reallocate for a larger buffer
261  deallocate(stbuffer)
262  buffersize = buffersize + default_buffer_size
263  allocate(stbuffer(buffersize))
264 
265  ! Copy back the original
266  stbuffer(1:k-1) = buffercopy
267  deallocate(buffercopy)
268  end if
269  if (j == 0) then
270  allocate(character(len = 0) :: stbuffer(k)%str)
271  else
272  allocate(character(len = j) :: stbuffer(k)%str)
273  stbuffer(k)%str = buffer(1:j)
274  end if
275  j = 0
276  end if
277  end do
278 
279  ! Catch the last string (between the final delimiter and end of txt)
280  if (j /= 0) then
281  k = k + 1
282  if (k > buffersize) then
283  ! Create a temporary copy of the string buffer
284  allocate(buffercopy(k - 1))
285  buffercopy = stbuffer
286 
287  ! Deallocate the string buffer, and then reallocate for a larger buffer
288  deallocate(stbuffer)
289  buffersize = buffersize + 1 ! We only need 1 more element
290  allocate(stbuffer(buffersize))
291 
292  ! Copy back the original
293  stbuffer(1:k-1) = buffercopy
294  deallocate(buffercopy)
295  end if
296  allocate(character(len = j) :: stbuffer(k)%str)
297  stbuffer(k)%str = buffer(1:j)
298  end if
299 
300  ! End
301  allocate(rst(k))
302  rst = stbuffer(1:k)
303  end function

361  function split_string_by_char(txt, delim, buffer, def, iwork) result(n)
362  ! Arguments
363  character(len = *), intent(in) :: txt
364  character, intent(in) :: delim
365  type(string), intent(out), dimension(:), target :: buffer
366  integer(int32), intent(in), optional :: def
367  integer(c_size_t), intent(out), optional, dimension(:), target :: iwork
368  integer(int32) :: n
369 
370  ! Parameters
371  integer(c_size_t), parameter :: def_size = 1024
372 
373  ! Local Variables
374  type(c_ptr) :: bfr(size(buffer))
375  integer(c_size_t) :: i, nbuffers, buffersize, nstr
376  character(kind = c_char) :: splitter
377  integer(c_size_t), allocatable, target, dimension(:) :: iw
378  integer(c_size_t), pointer, dimension(:) :: wptr
379 
380  ! Initialization
381  nbuffers = int(size(buffer), c_size_t)
382  if (present(def)) then
383  buffersize = int(def, c_size_t)
384  else
385  buffersize = def_size
386  end if
387  do i = 1, nbuffers
388  if (allocated(buffer(i)%str)) then
389  if (len(buffer(i)%str) /= buffersize) deallocate(buffer(i)%str)
390  allocate(character(len = bufferSize) :: buffer(i)%str)
391  else
392  allocate(character(len = bufferSize) :: buffer(i)%str)
393  end if
394  bfr(i) = c_loc(buffer(i)%str)
395  end do
396  splitter = delim
397 
398  ! Establish the workspace array
399  if (present(iwork)) then
400  if (size(iwork) >= nbuffers) then
401  wptr => iwork(1:nbuffers)
402  else
403  allocate(iw(nbuffers))
404  wptr => iw
405  end if
406  else
407  allocate(iw(nbuffers))
408  wptr => iw
409  end if
410 
411  ! Split the string
412  call split_string_delim(txt // c_null_char, splitter, bfr, nbuffers, &
413  buffersize, nstr, iw)
414  n = int(nstr, int32)
415 
416  ! Remove the c_null_char from the end of each string
417  do i = 1, nstr
418  nstr = wptr(i)
419  buffer(i)%str(nstr+1:buffersize) = ""
420  end do
421  end function
