828  pure function quadratic_cost_function(n, a, y) result(c)
829  ! Arguments
830  integer(int32), intent(in) :: n
831  real(real64), intent(in), dimension(:) :: a, y
832  real(real64) :: c
833 
834  ! Local Variables
835  integer(int32) :: i, m
836 
837  ! Process
838  m = min(size(a), size(y))
839  c = 0.0d0
840  do i = 1, m
841  c = c + (y(i) - a(i))**2
842  end do
843  c = c / (2.0d0 * real(n, real64))
844  end function

853  pure function cross_entropy_cost_function(n, a, y) result(c)
854  ! Arguments
855  integer(int32), intent(in) :: n
856  real(real64), intent(in), dimension(:) :: a, y
857  real(real64) :: c
858 
859  ! Local Variables
860  integer(int32) :: i, m
861 
862  ! Process
863  m = min(size(a), size(y))
864  c = 0.0d0
865  do i = 1, m
866  c = c + y(i) * log(a(i)) + (1.0d0 - y(i)) * log(1.0d0 - a(i))
867  end do
868  c = -c / real(n, real64)
869  end function
