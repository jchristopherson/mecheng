51 module subroutine cal_set_capacity(this, n, err)
52  ! Arguments
53  class(calibration), intent(inout) :: this
54  integer(int32), intent(in) :: n
55  class(errors), intent(inout), optional, target :: err
56 
57  ! Local Variables
58  class(errors), pointer :: errmgr
59  type(errors), target :: deferr
60  real(real64), allocatable, dimension(:,:) :: cpy
61  integer(int32) :: m, npts, flag
62 
63  ! Set up the error handler
64  if (present(err)) then
65  errmgr => err
66  else
67  errmgr => deferr
68  end if
69  npts = this%get_count()
70 
71  if (allocated(this%m_data)) then
72  ! Create a copy of any existing data - if necessary
73  if (npts > 0) then
74  allocate(cpy(npts, 2), stat = flag)
75  if (flag /= 0) go to 100
76  cpy = this%m_data(1:npts,:)
77  end if
78 
79  ! Free the existing memory
80  deallocate(this%m_data)
81  end if
82 
83  ! Reallocate to the correct size
84  allocate(this%m_data(n, 2), stat = flag)
85  if (flag /= 0) go to 100
86 
87  ! Copy back any data - if necessary
88  if (npts > 0) then
89  m = min(npts, n)
90  this%m_data(1:m,:) = cpy(1:m,:)
91  end if
92 
93  ! End
94  return
95 
96  ! Memory issue handling
97 100 continue
98  call errmgr%report_error("cal_set_capacity", &
