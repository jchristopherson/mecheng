52 module subroutine cal_set_capacity(this, n, err)
53  ! Arguments
54  class(calibration), intent(inout) :: this
55  integer(int32), intent(in) :: n
56  class(errors), intent(inout), optional, target :: err
57 
58  ! Local Variables
59  class(errors), pointer :: errmgr
60  type(errors), target :: deferr
61  real(real64), allocatable, dimension(:,:) :: cpy
62  integer(int32) :: m, npts, flag
63 
64  ! Set up the error handler
65  if (present(err)) then
66  errmgr => err
67  else
68  errmgr => deferr
69  end if
70  npts = this%get_count()
71 
72  if (allocated(this%m_data)) then
73  ! Create a copy of any existing data - if necessary
74  if (npts > 0) then
75  allocate(cpy(npts, 2), stat = flag)
76  if (flag /= 0) go to 100
77  cpy = this%m_data(1:npts,:)
78  end if
79 
80  ! Free the existing memory
81  deallocate(this%m_data)
82  end if
83 
84  ! Reallocate to the correct size
85  allocate(this%m_data(n, 2), stat = flag)
86  if (flag /= 0) go to 100
87 
88  ! Copy back any data - if necessary
89  if (npts > 0) then
90  m = min(npts, n)
91  this%m_data(1:m,:) = cpy(1:m,:)
92  end if
93 
94  ! End
95  return
96 
97  ! Memory issue handling
98 100 continue
99  call errmgr%report_error("cal_set_capacity", &
100  "Insufficient memory available.", &
101  cal_out_of_memory_error)
102 end subroutine
