33  module function signal_magnitude_win(x, winfun, winsize) result(y)
34  ! Arguments
35  real(real64), intent(in), dimension(:) :: x
36  procedure(window_function), pointer, intent(in) :: winfun
37  integer(int32), intent(in) :: winsize
38  real(real64), allocatable, dimension(:) :: y
39 
40  ! Parameters
41  real(real64), parameter :: zero = 0.0d0
42  real(real64), parameter :: two = 2.0d0
43 
44  ! Local Variables
45  integer(int32) :: i, j, lwork, lwsave, flag, nxfrm
46  real(real64), allocatable, dimension(:,:) :: bf
47  real(real64), dimension(winsize) :: w
48  real(real64), allocatable, dimension(:) :: wsave, work
49  real(real64) :: winsum
50  complex(real64) :: num
51 
52  ! Initialization
53  lwork = winsize
54  lwsave = winsize + int(log(real(winsize, real64)) / log(two), int32) + 4
55  allocate(wsave(lwsave))
56  allocate(work(lwork))
57  call rfft1i(winsize, wsave, lwsave, flag)
58  if (mod(winsize, 2) == 0) then
59  nxfrm = winsize / 2 + 1
60  else
61  nxfrm = (winsize + 1) / 2
62  end if
63  allocate(y(nxfrm))
64  y = zero
65 
66  ! Buffer the data, and compute the window function
67  bf = buffer(x, winsize)
68  winsum = zero
69  do i = 1, winsize
70  w(i) = winfun(i - 1, winsize)
71  winsum = winsum + w(i)
72  end do
73  winsum = winsum / winsize
74 
75  ! Apply the window function, and compute the FFT of each buffer
76  do i = 1, size(bf, 2)
77  ! Apply the window function
78  bf(:,i) = w * bf(:,i)
79 
80  ! Compute the FFT - only the positive half
81  call rfft1f(winsize, 1, bf(:,i), winsize, wsave, lwsave, work, &
82  lwork, flag)
83 
84  ! Compute the magnitude
85  if (mod(winsize, 2) == 0) then
86  y(1) = y(1) + abs(bf(1, i) / winsum)
87  do j = 2, nxfrm - 1
88  num = cmplx(bf(2*j-2,i), bf(2*j-1,i), real64)
89  y(j) = y(j) + abs(num / winsum)
90  end do
91  y(nxfrm) = y(i) + abs(bf(winsize, i) / winsum)
92  else
93  y(1) = y(1) + abs(bf(1, i)) / winsum
94  do j = 2, nxfrm
95  num = cmplx(bf(2*j-2,i), bf(2*j-1,i), real64)
96  y(j) = y(j) + abs(num / winsum)
97  end do
98  end if
99  end do
100 
101  ! Compute the average
102  y = y / size(bf, 2)
103  end function
