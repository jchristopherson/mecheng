250  pure function plane_from_3_points(p1, p2, p3) result(p)
251  ! Arguments
252  real(real64), intent(in), dimension(3) :: p1, p2, p3
253  type(plane) :: p
254 
255  ! Local Variables
256  real(real64), dimension(3) :: n
257 
258  ! Compute the normal vector of the plane
259  n = cross(p2 - p1, p3 - p1)
260  n = n / norm2(n)
261 
262  p%a = n(1)
263  p%b = n(2)
264  p%c = n(3)
265 
266  ! Treat point 1 as the origin, and then "d" can be found

447  ! Local Variables

450  real(real64), intent(in) :: angle

452  type(plane) :: p
453 
454  ! Local Variables
455  type(quaternion) :: q
456  real(real64), dimension(3) :: rpt
457 
458  ! Construct a quaternion using the supplied angle and axis information
459  call q%from_angle_axis(angle, axis)
460 
461  ! Normalize the quaternion to ensure we're dealing with a unit
462  ! quaternion
463  call q%normalize()
464 
465  ! Apply the rotation to the point. After this, we'll have a line that
466  ! defines 2 points, and a 3rd point to define the plane.
467  rpt = q%transform(pt)
468 
469  ! Define the plane from the 3 points we now have
470  if (present(opt)) then
471  p = plane_from_3_points(opt, axis + opt, rpt)
472  else

474  end if
