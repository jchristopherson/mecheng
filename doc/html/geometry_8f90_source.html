281  type(plane) :: p
282

284  real(real64), dimension(3) :: n
285 
286  ! Compute the normal vector of the plane
287  n = cross(p2 - p1, p3 - p1)
288  if (n < 2.0d0 * epsilon(2.0d0)) then

302  p%d = -(n(1) * p1(1) + n(2) * p1(2) + n(3) * p1(3))

505  function line_from_2_planes(p1, p2) result(r)
506  ! Arguments
507  use linalg_immutable
508  class(plane) :: p1, p2
509  type(line) :: r
510 
511  ! Local Variables
512  real(real64) :: tol
513  real(real64), dimension(3) :: n1, n2, n1n2, pt
514  real(real64), dimension(2) :: d
515  real(real64), dimension(2, 3) :: c
516 
517  ! Compute the normal vectors of each plane
518  n1 = p1%normal_vector()
519  n2 = p2%normal_vector()
520 
521  ! Ensure the planes are not parallel
522  tol = 2.0d0 * epsilon(tol)
523  n1n2 = cross(n1, n2)
524  if (norm2(n1n2) < tol) then
525  ! The two planes are parallel
526  r%a = 0.0d0
527  r%b = 0.0d0
528  return
529  end if
530 
531  ! Notice, N1 cross N2 is parallel to the line of intersection between
532  ! the two planes. As such, the problem now is to find a point that
533  ! lies on both planes. This may be achieved by solving the two
534  ! plane equations simultaneously.
535  c = reshape([p1%a, p2%a, p1%b, p2%b, p1%c, p2%c], [2, 3])
536  d = -[p1%d, p2%d]
537 
538  ! Solve the system of equations via a least-squares technique
539  pt = matmul(mat_pinverse(c), d)
540 
541  ! Now that we have a point on the line, and the direction of the line,
542  ! we know the line
543  r = line_from_2_points(pt, n1n2 + pt)
544  end function

688  ! Arguments

692

695  real(real64), dimension(3) :: cp, u, v
696  real(real64) :: s, t, denom
697 
698  ! Initialization
699  u = ln2%direction()
700  v = ln1%direction()
701  cp = cross(u, v)
702

704  a1(:,2) = u
