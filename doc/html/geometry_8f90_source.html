465  function line_from_2_planes(p1, p2) result(r)
466  ! Arguments
467  use linalg_immutable
468  class(plane) :: p1, p2
469  type(line) :: r
470 
471  ! Local Variables
472  real(real64) :: tol
473  real(real64), dimension(3) :: n1, n2, n1n2, pt
474  real(real64), dimension(2) :: d
475  real(real64), dimension(2, 3) :: c
476 
477  ! Compute the normal vectors of each plane
478  n1 = p1%normal_vector()
479  n2 = p2%normal_vector()
480 
481  ! Ensure the planes are not parallel
482  tol = 2.0d0 * epsilon(tol)
483  n1n2 = cross(n1, n2)
484  if (norm2(n1n2) < tol) then
485  ! The two planes are parallel
486  r%a = 0.0d0
487  r%b = 0.0d0
488  return
489  end if
490 
491  ! Notice, N1 cross N2 is parallel to the line of intersection between
492  ! the two planes. As such, the problem now is to find a point that
493  ! lies on both planes. This may be achieved by solving the two
494  ! plane equations simultaneously.
495  c = reshape([p1%a, p2%a, p1%b, p2%b, p1%c, p2%c], [2, 3])
496  d = -[p1%d, p2%d]
497 
498  ! Solve the system of equations via a least-squares technique
499  pt = matmul(mat_pinverse(c), d)
500 
501  ! Now that we have a point on the line, and the direction of the line,
502  ! we know the line
503  r = line_from_2_points(pt, n1n2 + pt)
504  end function
