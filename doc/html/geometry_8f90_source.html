286  ! Arguments
287  real(real64), intent(in), dimension(3) :: p1, p2, p3
288  type(plane) :: p
289 
290  ! Local Variables
291  real(real64), dimension(3) :: n
292 
293  ! Compute the normal vector of the plane
294  n = cross(p2 - p1, p3 - p1)
295  if (norm2(n) < 2.0d0 * epsilon(2.0d0)) then
296  p%a = 0.0d0
297  p%b = 0.0d0
298  p%c = 0.0d0
299  p%d = 0.0d0
300  return
301  end if
302 
303  p%a = n(1)
304  p%b = n(2)
305  p%c = n(3)
306 
307  ! Treat point 1 as the origin, and then "d" can be found
308  p%d = -(n(1) * p1(1) + n(2) * p1(2) + n(3) * p1(3))
309  end function

692  function line_from_2_planes(p1, p2) result(r)
693  ! Arguments
694  use linalg_immutable
695  class(plane) :: p1, p2
696  type(line) :: r
697 
698  ! Local Variables
699  real(real64) :: tol
700  real(real64), dimension(3) :: n1, n2, n1n2, pt
701  real(real64), dimension(2) :: d
702  real(real64), dimension(2, 3) :: c
703 
704  ! Compute the normal vectors of each plane
705  n1 = p1%normal_vector()
706  n2 = p2%normal_vector()
707 
708  ! Ensure the planes are not parallel
709  tol = 2.0d0 * epsilon(tol)
710  n1n2 = cross(n1, n2)
711  if (norm2(n1n2) < tol) then
712  ! The two planes are parallel
713  r%a = 0.0d0
714  r%b = 0.0d0
715  return
716  end if
717 
718  ! Notice, N1 cross N2 is parallel to the line of intersection between
719  ! the two planes. As such, the problem now is to find a point that
720  ! lies on both planes. This may be achieved by solving the two
721  ! plane equations simultaneously.
722  c = reshape([p1%a, p2%a, p1%b, p2%b, p1%c, p2%c], [2, 3])
723  d = -[p1%d, p2%d]
724 
725  ! Solve the system of equations via a least-squares technique
726  pt = matmul(mat_pinverse(c), d)
727 
728  ! Now that we have a point on the line, and the direction of the line,
729  ! we know the line
730  r = line_from_2_points(pt, n1n2 + pt)
731  end function

970  function shortest_line_to_line(ln1, ln2) result(r)
971  ! Arguments
972  use linalg_core
973  class(line), intent(in) :: ln1, ln2
974  type(line) :: r
975 
976  ! Local Variables
977  real(real64), dimension(3, 3) :: a1, a2
978  real(real64), dimension(3) :: cp, u, v
979  real(real64) :: s, t, denom
980 
981  ! Initialization
982  u = ln2%direction()
983  v = ln1%direction()
984  cp = cross(u, v)
985 
986  a1(:,1) = ln1%a - ln2%a
987  a1(:,2) = u
988  a1(:,3) = cp
989 
990  a2(:,1) = a1(:,1)
991  a2(:,2) = v
992  a2(:,3) = cp
993 
994  denom = norm2(cp)**2
995 
996  ! Compute the parameter on each line
997  s = det(a1) / denom
998  t = det(a2) / denom
999 
1000  ! Compute the new line as we now have 2 points
1001  r = line_from_2_points(ln1%evaluate(s), ln2%evaluate(t))
1002  end function
