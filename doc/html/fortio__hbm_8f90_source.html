98  function hbm_data_to_matrix(x, err) result(rst)
99  ! Arguments
100  class(hbm_data_file), intent(in) :: x
101  class(errors), intent(inout), optional, target :: err
102  real(real64), allocatable, dimension(:,:) :: rst
103 
104  ! Local Variables
105  integer(int32) :: j, m, n, chanlength, info

107  class(errors), pointer :: errmgr
108 
109  ! Initialization
110  if (present(err)) then
111  errmgr => err

113  errmgr => deferr

116  m = 0
