45 ! Computes the range of an array of values.
46 pure function array_range(x) result(r)
47  ! Arguments
48  real(real64), intent(in), dimension(:) :: x
49  real(real64) :: r
50 
51  ! Local Variables
52  integer(int32) :: i, n
53  real(real64) :: maxx, minx
54 
55  ! Quick Return
56  n = size(x)
57  r = 0.0d0
58  if (n == 0) return
59 
60  ! Determine the difference between the max and min values
61  maxx = x(1)
62  minx = x(1)
63  do i = 2, n
64  if (x(i) > maxx) maxx = x(i)
65  if (x(i) < minx) minx = x(i)
66  end do
67  r = maxx - minx

71 module function emp_gauge_r_r(x, tol, kf, err) result(rst)
72  ! Arguments
73  real(real64), intent(in), dimension(:,:,:) :: x
74  real(real64), intent(in), optional :: tol, kf
75  class(errors), intent(inout), optional, target :: err
76  type(emp_grr_results) :: rst
77 
78  ! Local Variables
79  class(errors), pointer :: errmgr
80  type(errors), target :: deferr
81  character(len = 256) :: errmsg
82  integer(int32) :: i, j, k, npart, nrep, nop, flag
83  real(real64) :: d2rep, d2rpro, d2part
84  real(real64), allocatable, dimension(:) :: reprng
85 
86  ! Initialization
87  if (present(err)) then
88  errmgr => err
89  else
90  errmgr => deferr
91  end if
92  rst%scale = 6.0d0
93  if (present(kf)) rst%scale = kf
94 
95  npart = size(x, 1)
96  nrep = size(x, 2)
97  nop = size(x, 3)
98 
99  ! Input Check
100  if (npart < 2) then
101  write (errmsg, '(AI0A)') &
102  "Expected to find at least 2 parts, but only found ", &
103  npart, "."
104  call errmgr%report_error("emp_gauge_r_r", trim(errmsg), &
105  ms_invalid_data_set_error)
106  return
107  end if
108  if (nrep < 2) then
109  write (errmsg, '(AI0A)') &
110  "Expected to find at least 2 tests, but only found ", &
111  nrep, "."
112  call errmgr%report_error("emp_gauge_r_r", trim(errmsg), &
113  ms_invalid_data_set_error)
114  return
115  end if
116  if (nop < 2) then
117  write (errmsg, '(AI0A)') &
118  "Expected to find at least 2 operators, but only found ", &
119  nop, "."
120  call errmgr%report_error("emp_gauge_r_r", trim(errmsg), &
121  ms_invalid_data_set_error)
122  return
123  end if
124 
125  ! Compute the approriate bias correction factors
126  d2rep = bias_correction_factor(nrep)
127  d2rpro = bias_correction_factor(nop)
128  d2part = bias_correction_factor(npart)
129 
130  ! Output Memory Allocations
131  allocate(rst%operator_means(nop), stat = flag)
132  if (flag == 0) allocate(rst%part_means(npart), stat = flag)
133  if (flag == 0) allocate(rst%repeat_means(nrep), stat = flag)
134  if (flag == 0) allocate(reprng(npart * nop), stat = flag)
135  if (flag /= 0) then
136  call errmgr%report_error("emp_gauge_r_r", &
137  "Insufficient memory available.", &
138  ms_out_of_memory_error)
139  return
140  end if
141 
142  ! Compute the mean (average) terms
143  do k = 1, nop
144  rst%operator_means(k) = mean(reshape(x(:,:,k), [npart * nrep]))
145  end do
146 
147  do j = 1, nrep
148  rst%repeat_means(j) = mean(reshape(x(:,j,:), [npart * nop]))
149  end do
150 
151  do i = 1, npart
152  rst%part_means(i) = mean(reshape(x(i,:,:), [nrep * nop]))
153  end do
154 
155  ! Compute the range terms
156  rst%operator_range = array_range(rst%operator_means)
157  rst%part_range = array_range(rst%part_means)
158  j = 0
159  do k = 1, nop
160  do i = 1, npart
161  j = j + 1
162  reprng(j) = array_range(x(i,:,k))
163  end do
164  end do
165  rst%average_range = mean(reprng)
166 
167  ! Compute the variance terms
168  rst%repeatability = (rst%average_range / d2rep)**2
169  rst%reproducibility = (rst%operator_range / d2rpro)**2 - &
170  (nop / (npart * nop * nrep)) * rst%repeatability
171  rst%gauge_variation = rst%repeatability + rst%reproducibility
172  rst%part_variation = (rst%part_range / d2part)**2 - &
173  (npart / (npart * nop * nrep)) * rst%repeatability
174  rst%total_variation = rst%gauge_variation + rst%part_variation
175 
176  ! Compute the P/T information
177  if (present(tol)) then
178  rst%tolerance = tol
179  rst%pt_ratio = rst%scale * sqrt(rst%gauge_variation) / tol
180  else
181  rst%tolerance = 0.0d0
182  rst%pt_ratio = 0.0d0
183  end if
184 
185  ! Compute the P/TV information
186  rst%ptv_ratio = rst%gauge_variation / rst%total_variation
187 end function
