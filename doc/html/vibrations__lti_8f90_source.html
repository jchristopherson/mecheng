315  module subroutine lti_to_ss(this, ss, err)
316  ! Arguments
317  class(LTI), intent(in) :: this
318  class(state_space), intent(out) :: ss
319  class(errors), intent(inout), optional, target :: err
320 
321  ! Local Variables
322  integer(int32) :: i, j, n, nb
323  real(real64) :: a0, b0
324  class(errors), pointer :: errmgr
325  type(errors), target :: deferr
326 
327  ! Initialization
328  if (present(err)) then
329  errmgr => err
330  else
331  errmgr => deferr
332  end if
333 
334  ! Ensure the transfer function is valid
335  if (.not.this%validate()) then
336  call errmgr%report_error("lti_to_ss", &
337  "The transfer function is not valid.", &
338  mech_invalid_transfer_function_error)
339  return
340  end if
341 
342  ! Allocate space for the matrices
343  n = this%denominator%order()
344  allocate(ss%A(n, n))
345  allocate(ss%B(n, 1))
346  allocate(ss%C(1, n))
347  allocate(ss%D(1, 1))
348 
349  ! Construct A
350  do j = 2, n
351  do i = 1, n
352  if (i == j - 1) then
353  ss%A(i,j) = 1.0d0
354  else
355  ss%A(i,j) = 0.0d0
356  end if
357  end do
358  end do
359  a0 = this%denominator%get(n+1) ! Coefficient of the highest power
360  do i = 1, n
361  ss%A(i,1) = -(this%denominator%get(n-i+1)) / a0
362  end do
363 
364  ! Construct B
365  ss%B = 0.0d0
366  nb = this%numerator%order()
367  do i = 1, nb + 1
368  ss%B(n-i+1,1) = this%numerator%get(i)
369  end do
370 
371  if (nb == n) then
372  b0 = this%numerator%get(nb+1)
373  ss%B(:,1) = (ss%B(:,1) / b0) + ss%A(:,1) * b0
374  else
375  b0 = 0.0d0
376  end if
377 
378  ! Construct C
379  ss%C(1,1) = 1.0d0
380  ss%C(1,2:n) = 0.0d0
381 
382  ! Construct D
383  ss%D(1,1) = b0
384  end subroutine

411  if (k <= n) then

413  arg2 = abs(aimag(poles(k)))
414  if (abs(arg1 - arg2) < tol) then
415  j = j + 1

418  end if
419  end if

422  ! A complex-conjugate pair was found
423  i = i + 2
424  else

426  i = i + 1
427  end if
428  end do
429

437  ! frequency values, in Hz

439 
440  ! Compute the angle of each value from the negative real axis to
441  ! determine the damping ratio
442  x(:,2) = cos( atan2( abs(aimag(buffer(1:j))), abs(real(buffer(1:j))) ) )
