6  module function oe_step(this, fcn, x, y, xout, rtol, atol, err) result(brk)
7  ! Arguments
8  class(ode_euler), intent(inout) :: this
9  class(ode_helper), intent(inout) :: fcn
10  real(real64), intent(inout) :: x
11  real(real64), intent(inout), dimension(:) :: y
12  real(real64), intent(in) :: xout
13  real(real64), intent(in), dimension(:) :: rtol, atol
14  class(errors), intent(inout), optional, target :: err
15  logical :: brk
16 
17  ! Local Variables
18  real(real64) :: h, xerr
19 
20  ! Initialization
21  brk = .false.
22  h = xout - x
23  call this%initialize(fcn%get_equation_count(), err)
24 
25  ! Check to see if root finding is requested
26 
27  ! Process
28  if (this%m_first) call fcn%evaluate_ode(x, y, this%m_dydx)
29  this%m_dydx = h * this%m_dydx + this%m_summationError
30  this%m_y1 = y + this%m_dydx
31 
32  ! Update the summation error tracking
33  this%m_summationError = this%m_dydx - (this%m_y1 - y)
34 
35  ! Compute the function value at x + h
36  call fcn%evaluate_ode(xout, y, this%m_dydx1)
37 
38  ! Estimate the solution error
39  this%m_error = 0.5d0 * h * (this%m_dydx - this%m_dydx1)
40 
41  ! Update the end time and store the most recent function evaluation
42  x = xout
43  y = this%m_y1
44  this%m_dydx = this%m_dydx1
45  this%m_first = .false.
46  end function

95  int_out_of_memory_error)
