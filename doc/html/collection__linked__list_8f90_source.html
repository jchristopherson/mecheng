374  ! Arguments

378

382  type(errors), target :: deferr

384 
385  ! Initialization
386  if (present(err)) then
387  errmgr => err
388  else
389  errmgr => deferr
390  end if
391 
392  ! Process
393  allocate(node, stat = flag)
394  if (flag /= 0) then
395  call errmgr%report_error("ll_push", &
396  "Insufficient memory available.", &
397  collection_out_of_memory_error)
398  return
399  end if
400  node%item => x
401  nullify(node%next)
402  if (associated(this%m_last)) then
403  ! There are already nodes in the collection, and as such,
404  ! this is the new last node
405  this%m_last%next => node
406  node%previous => this%m_last
407  this%m_last => node
408  else
409  ! There are not any nodes in the collection. This node is both
410  ! the first, and the last node in the collection.
411  nullify(node%previous)
412  this%m_first => node
413  this%m_last => node
414  this%m_node => node
415  end if
416  this%m_count = this%m_count + 1
417  end subroutine

461  subroutine ll_insert(this, i, x, err)
462  ! Arguments
463  class(linked_list), intent(inout) :: this
464  integer(int32), intent(in) :: i
465  class(*), intent(in), target :: x
466  class(errors), intent(inout), optional, target :: err
467 
468  ! Local Variables
469  integer(int32) :: j, n
470  class(errors), pointer :: errmgr
471  type(errors), target :: deferr
472  type(dll_node), pointer :: ptr, prev, node
473  integer(int32) :: flag
474 
475  ! Initialization
476  n = this%get_count()
477  if (present(err)) then
478  errmgr => err
479  else
480  errmgr => deferr
481  end if
482 
483  ! Input Check
484  if (i <= 0 .or. i > n + 1) then
485  call errmgr%report_error("ll_insert", &
486  "The supplied index is outside the bounds of the list.", &
487  collection_invalid_input_error)
488  return
489  end if
490 
491  ! Find the right node
492  if (i == 1) then
493  ! We've got a new first node
494  allocate(node, stat = flag)
495  if (flag /= 0) then
496  call errmgr%report_error("ll_insert", &
497  "Insufficient memory available.", &
498  collection_out_of_memory_error)
499  return
500  end if
501  node%item => x
502  node%next => this%m_first
503  nullify(node%previous)
504  if (n > 0) then
505  this%m_first%previous => node
506  this%m_first => node
507  end if
508  else if (i == n + 1) then
509  ! Just push a node onto the end of the list
510  allocate(node, stat = flag)
511  if (flag /= 0) then
512  call errmgr%report_error("ll_insert", &
513  "Insufficient memory available.", &
514  collection_out_of_memory_error)
515  return
516  end if
517  node%item => x
518  nullify(node%next)
519  if (associated(this%m_last)) then
520  ! There are already nodes in the collection
521  this%m_last%next => node
522  node%previous => this%m_last
523  this%m_last => node
524  else
525  ! This is the first and only node in the collection
526  nullify(node%previous)
527  this%m_first => node
528  this%m_last => node
529  this%m_node => node
530  end if
531  else
532  ! We're somewhere in the middle
533  allocate(node, stat = flag)
534  if (flag /= 0) then
535  call errmgr%report_error("ll_insert", &
536  "Insufficient memory available.", &
537  collection_out_of_memory_error)
538  return
539  end if
540  ptr => this%m_first
541  do j = 2, i
542  prev => ptr
543  ptr => ptr%next
544  end do
545  node%item => x
546  node%next => ptr
547  node%previous => prev
548  prev%next => node
549  ptr%previous => node
550  end if
551  this%m_count = this%m_count + 1
552  end subroutine

567  subroutine ll_remove(this, i, err)
568  ! Arguments
569  class(linked_list), intent(inout) :: this
570  integer(int32), intent(in) :: i
571  class(errors), intent(inout), optional, target :: err
572 
573  ! Local Variables
574  integer(int32) :: j, n
575  class(errors), pointer :: errmgr
576  type(errors), target :: deferr
577  type(dll_node), pointer :: ptr, next, prev
578 
579  ! Initialization
580  n = this%get_count()
581  if (present(err)) then
582  errmgr => err
583  else
584  errmgr => deferr
585  end if
586 
587  ! Ensure there are items to remove
588  if (n == 0) then
589  call errmgr%report_error("ll_remove", &
590  "Attempted to remove items from an already empty list.", &
591  collection_invalid_operation_error)
592  return
593  end if
594 
595  ! Input Check
596  if (i <= 0 .or. i > n) then
597  call errmgr%report_error("ll_remove", &
598  "The supplied index is outside the bounds of the list.", &
599  collection_invalid_input_error)
600  return
601  end if
602 
603  ! Process
604  if (i == 1) then
605  ! Remove the first node
606  ptr => this%m_first
607  next => ptr%next
608  deallocate(ptr)
609  nullify(ptr)
610  if (associated(next)) then
611  nullify(next%previous)
612  this%m_first => next
613  else
614  nullify(this%m_first)
615  nullify(this%m_last)
616  end if
617  else if (i == n) then
618  ! Remove the last node
619  ptr => this%m_last
620  prev => ptr%previous
621  deallocate(ptr)
622  nullify(ptr)
623  if (associated(prev)) then
624  nullify(prev%next)
625  this%m_last => prev
626  else
627  nullify(this%m_first)
628  nullify(this%m_last)
629  end if
630  else
631  ! Remove something in the middle
632  ptr => this%m_first
633  do j = 2, i
634  prev => ptr
635  ptr => ptr%next
636  end do
637  next => ptr%next
638  prev%next => next
639  next%previous => prev
640  deallocate(ptr)
641  nullify(ptr)
642  end if
643  this%m_count = this%m_count - 1
644  end subroutine
