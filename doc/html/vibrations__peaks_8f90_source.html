7 module function frf_peak_detect_1(freq, amp, ranges, err) result(rst)
8  ! Arguments
9  real(real64), intent(in), dimension(:) :: freq, amp
10  class(frf_search_info), intent(in), dimension(:) :: ranges
11  class(errors), intent(inout), optional, target :: err
12  type(peak_info), allocatable, dimension(:) :: rst
13 
14  ! Local Variables
15  integer(int32) :: i, start, finish, nranges, flag
16  class(errors), pointer :: errmgr
17  type(errors), target :: deferr
18 
19  ! Initialization
20  if (present(err)) then
21  errmgr => err
22  else
23  errmgr => deferr
24  end if
25  nranges = size(ranges)
26 
27  ! Quick Return
28  if (size(freq) < 2) return
29  if (nranges == 0) return
30 
31  ! Ensure freq is monotonically increasing
32  if (.not.is_monotonic(freq)) then
33  ! ERROR: The array is not monotonic
34  call errmgr%report_error("frf_peak_detect_1", &
35  "The frequency array is not monotonic.", &
36  mech_nonmonotonic_error)
37  return
38  end if
39  if (freq(2) < freq(1)) then
40  ! ERROR: The array is not monotonically increasing
41  call errmgr%report_error("frf_peak_detect_1", &
42  "The frequency array is not monotonically increasing.", &
43  mech_nonmonotonic_error)
44  return
45  end if
46 
47  ! Ensure freq & amp are the same size
48  if (size(freq) /= size(amp)) then
49  ! ERROR: Array sizes must match
50  call errmgr%report_error("frf_peak_detect_1", &
51  "The frequency and amplitude arrays must be the same size.", &
52  mech_array_size_error)
53  return
54  end if
55 
56  ! Allocate space for the output
57  allocate(rst(nranges), stat = flag)
58  if (flag /= 0) then
59  ! ERROR: Memory issue
60  call errmgr%report_error("frf_peak_detect_1", &
61  "Insufficient memory available.", &
62  mech_out_of_memory_error)
63  return
64  end if
65 
66  ! Cycle over each range
67  do i = 1, nranges
68  ! Locate the start and finish indices
69  start = find_first_index(freq, ranges(i)%min_frequency)
70  finish = find_first_index(freq, ranges(i)%max_frequency)
71 
72  ! Search for max and min values over the requested range
73  rst(i) = peak_detect(amp(start:finish), &
74  ranges(i)%amplitude_threshold, errmgr)
75  if (errmgr%has_error_occurred()) return
76 
77  ! Offset the indices to account for the starting point
78  rst(i)%max_value_indices = rst(i)%max_value_indices + start - 1
79  rst(i)%min_value_indices = rst(i)%min_value_indices + start - 1
80  end do
81 end function

106 ! Finds the index of the first occurrence of the specified value. If
107 ! the value isn't found, the first index of the closest value larger
108 ! in magnitude. It is assumed that x is monotonically increasing.
109 pure function find_first_index(x, val) result(ind)
110  ! Arguments
111  real(real64), intent(in), dimension(:) :: x
112  real(real64), intent(in) :: val
113  integer(int32) :: ind
114 
115  ! Process
116  ind = 1
117  do while (x(ind) < val)

120  ind = size(x)

122  end if
123  end do
