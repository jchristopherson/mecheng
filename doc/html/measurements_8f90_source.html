719  use curvefit_statistics

721 
722  ! Arguments
723  real(real64), intent(in), dimension(:,:,:) :: x
724  class(plot), intent(inout) :: plt
725  class(errors), intent(inout), optional, target :: err
726 
727  ! Local Variables
728  class(errors), pointer :: errmgr
729  type(errors), target :: deferr
730  type(plot_data_2d) :: pdata
731  type(plot_data_error_bars) :: edata
732  integer(int32) :: i, k, nparts, nops, flag
733  real(real64), allocatable, dimension(:) :: xd, y, ymin, ymax
734  type(color) :: clr
735 
736  ! Initialization
737  if (present(err)) then
738  errmgr => err
739  else
740  errmgr => deferr
741  end if
742  nparts = size(x, 1)
743  nops = size(x, 3)
744  allocate(xd(nparts), stat = flag)
745  if (flag == 0) allocate(y(nparts), stat = flag)
746  if (flag == 0) allocate(ymin(nparts), stat = flag)
747  if (flag == 0) allocate(ymax(nparts), stat = flag)
748  if (flag /= 0) then
749  call errmgr%report_error("plot_grr", &
750  "Insufficient memory available.", ms_out_of_memory_error)
751  return
752  end if
753 
754  ! Establish an x-axis
755  do i = 1, nparts
756  xd(i) = real(i)
757  end do
758 
759  ! Cycle over each operator, and generate the plot data
760  do k = 1, nops
761  ! Compute the mean of each parts set, and determine the range
762  do i = 1, nparts
763  y(i) = mean(x(i,:,k))
764  ! yerr(i) = 0.5d0 * (maxval(x(i,:,k)) - minval(x(i,:,k)))
765  ymin(i) = minval(x(i,:,k))
766  ymax(i) = maxval(x(i,:,k))
767  end do
768 
769  ! Define the plot data
770  call pdata%define_data(xd, y)
771  call pdata%set_line_style(line_dotted)
772  call plt%push(pdata)
773  clr = pdata%get_line_color()
774 
775  call edata%define_y_error_data(xd, y, ymin, ymax)
776  call edata%set_name("Operator " // to_string(k))
777  call edata%set_line_color(clr)
778  call plt%push(edata)
779 
780  ! Notice, the above code works as a copy of each data object is
781  ! stored and managed by the plot
782  end do
