224  function radial_distance_2d(x, y, tol, err) result(pts)
225  ! Arguments
226  real(real64), intent(in), dimension(:) :: x, y
227  real(real64), intent(in) :: tol
228  class(errors), intent(inout) :: err
229  real(real64), allocatable, dimension(:,:) :: pts
230 
231  ! Local Variables
232  integer(int32) :: i, j, n, nvalid, flag
233  logical, allocatable, dimension(:) :: valid
234  real(real64) :: r, xref, yref
235 
236  ! Initialization
237  n = size(x)
238  if (n == 0) return
239  i = 2
240  xref = x(1)
241  yref = y(1)
242  nvalid = 1
243 
244  ! Local Memory Allocation
245  allocate(valid(n), stat = flag)
246  if (flag /= 0) then
247  call err%report_error("radial_distance_2d", &
248  "Insufficient memory available.", &
249  plot_out_of_memory_error)
250  return
251  end if
252  valid(1) = .true.
253 
254  ! Cycle through and determine which points to keep
255  do
256  if (i > n) exit
257  r = pythag2(x(i), y(i), xref, yref)
258  if (r < tol) then
259  ! The point is too close, reject it
260  valid(i) = .false.
261  else
262  ! The point is outside the tolerance, and is OK
263  valid(i) = .true.
264  nvalid = nvalid + 1
265 
266  ! Move the reference point
267  xref = x(i)
268  yref = y(i)
269  end if
270  i = i + 1
271  end do
272 
273  ! Allocate space, and collect all valid points
274  allocate(pts(nvalid, 2), stat = flag)
275  if (flag /= 0) then
276  call err%report_error("radial_distance_2d", &
277  "Insufficient memory available.", &
278  plot_out_of_memory_error)
279  return
280  end if
281  j = 1
282  do i = 1, n
283  if (valid(i)) then
284  pts(j,1) = x(i)
285  pts(j,2) = y(i)
286  j = j + 1
287  end if
288  end do

362  pure function pythag2(x, y, xo, yo) result(r)
363  ! Arguments
364  real(real64), intent(in) :: x, y, xo, yo
365  real(real64) :: r
366 
367  ! Local Variables
368  real(real64) :: w, xabs, yabs
369 
370  ! Process
371  xabs = abs(x - xo)
372  yabs = abs(y - yo)
373  w = max(xabs, yabs)
374  if (w < epsilon(w)) then
375  r = xabs + yabs
376  else
377  r = w * sqrt((xabs / w)**2 + (yabs / w)**2)
378  end if
379  end function

381  pure function pythag3(x, y, z, xo, yo, zo) result(r)
382  ! Arguments
383  real(real64), intent(in) :: x, y, z, xo, yo, zo
384  real(real64) :: r
385 
386  ! Local Variables
387  real(real64) :: w, xabs, yabs, zabs
388 
389  ! Process
390  xabs = abs(x - xo)
391  yabs = abs(y - yo)
392  zabs = abs(z - zo)
393  w = max(xabs, yabs, zabs)
394  if (w < epsilon(w)) then
395  r = xabs + yabs + zabs
396  else
397  r = w * sqrt((xabs / w)**2 + (yabs / w)**2 + (zabs / w)**2)
