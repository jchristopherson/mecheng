87  ! Ensure the input arrays are properly sized

102  return

105  ! Ensure period is a positive, non-zero value
106  if (period < epsilon(period)) then
107  call errmgr%report_error("compute_poincare_section", &
108  "The sampling period must be a positive value larger " // &
109  "than machine precision.", mech_invalid_input_error)
110  return
111  end if
112 
113  ! Build the time array at which values are to be sampled
114  maxt = maxval(t)
115  mint = minval(t)
116  nsample = int((maxt - mint) / period, int32) + 1
117  allocate(samples(nsample), stat = flag)
118  if (flag == 0) allocate(rst(nsample, 2), stat = flag)
119  if (flag /= 0) then
120  call errmgr%report_error("compute_poincare_section", &
121  "Insufficient memory available.", &
122  mech_out_of_memory_error)
123  return
124  end if
125 
126  samples(1) = mint
127  do i = 2, nsample
128  samples(i) = samples(i-1) + period
129  end do
130 
131  ! Interpolate to obtain the values - use linear interpolation
132  call interp%initialize(t, x, err = errmgr)
133  if (errmgr%has_error_occurred()) return
134  rst(:,1) = interp%interpolate(samples)
135 
136  call interp%initialize(t, v, err = errmgr)
137  if (errmgr%has_error_occurred()) return
138  rst(:,2) = interp%interpolate(samples)

209  if (flag == 0) allocate(vecs(n,n), stat = flag)

223  end do
224  end do

228  if (errmgr%has_error_occurred()) return
229 
230  ! Sort the eigenvalues and eigenvectors
231  call sort(vals, vecs, .true.)
232 
233  ! Report the results
234  do i = 1, n
235  rst(i)%frequency = sqrt(real(vals(i), real64)) / (2.0d0 * pi)
236  rst(i)%mode_shape = real(vecs(:,i), real64)
237 
238  ! Normalize the mode shape vector to it's largest magnitude component
239  ind = idamax(n, rst(i)%mode_shape, 1)
240  val = rst(i)%mode_shape(ind)
241  rst(i)%mode_shape = rst(i)%mode_shape / val
242  end do
